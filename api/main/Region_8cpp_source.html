130 MxlRegions mxlFabricsRegionsFromMutableFlow(FlowData& flow)
131 {
132 auto mxlRegions = mxlFabricsRegionsFromFlow(flow);
133
134 if (mxlIsDiscreteDataFormat(static_cast<int>(flow.flowInfo()->config.common.format)))
135 {
136 auto& discreteFlow = static_cast<DiscreteFlowData&>(flow);
137
138 if (mxlRegions._regions.size() != discreteFlow.grainCount())
139 {
140 throw Exception::invalidState("Unexpected number of grains in discrete flow");
141 }
142
143 for (std::size_t i = 0; i < discreteFlow.grainCount(); ++i)
144 {
145 mxlRegions._regions[i].grainIndexPtr = &discreteFlow.grainAt(i)->header.info.index;
146 mxlRegions._regions[i].validSlicesPtr = &discreteFlow.grainAt(i)->header.info.validSlices;
147 }
148 }
149
150 return mxlRegions;
151 }

153 MxlRegions mxlFabricsRegionsFromFlow(FlowData const& flow)
154 {
155 static_assert(sizeof(GrainHeader) == 8192,
156 "GrainHeader type size changed! The Fabrics API makes assumptions on the memory layout of a flow, please review the code below if the "
157 "change is intended!");
158
159 if (flow.flowInfo()->config.common.payloadLocation != MXL_PAYLOAD_LOCATION_HOST_MEMORY)
160 {
161 throw Exception::make(MXL_ERR_UNKNOWN,
162 "GPU memory is not currently supported in the Flow API of MXL. Edit the code below when it is supported");
163 }
164
165 if (mxlIsDiscreteDataFormat(static_cast<int>(flow.flowInfo()->config.common.format)))
166 {
167 auto const& discreteFlow = static_cast<DiscreteFlowData const&>(flow);
168 auto regions = std::vector<Region>{};
169 regions.reserve(discreteFlow.grainCount());
170 for (auto i = std::size_t{0}; i < discreteFlow.grainCount(); ++i)
171 {
172 auto grain = discreteFlow.grainAt(i);
173
174 auto grainInfoBaseAddr = reinterpret_cast<std::uintptr_t>(discreteFlow.grainAt(i));
175 auto grainInfoSize = sizeof(GrainHeader);
176 auto grainPayloadSize = grain->header.info.grainSize;
177
178 regions.emplace_back(grainInfoBaseAddr, grainInfoSize + grainPayloadSize, nullptr, nullptr, Region::Location::host());
179 }
180
181 return {std::move(regions),
182 DataLayout::fromDiscrete(std::to_array(discreteFlow.flowInfo()->config.discrete.sliceSizes)),
183 discreteFlow.flowInfo()->config.common.maxSyncBatchSizeHint};
184 }
185 else if (mxlIsContinuousDataFormat(static_cast<int>(flow.flowInfo()->config.common.format)))
186 {
187 auto const& continuousFlow = static_cast<ContinuousFlowData const&>(flow);
188 auto regions = std::vector<Region>{};
189
190 // For the continuous flow, the data layout is a single contiguous buffer
191 regions.emplace_back(reinterpret_cast<std::uintptr_t>(continuousFlow.channelData()),
192 continuousFlow.channelDataSize(),
193 nullptr,
194 nullptr,
195 Region::Location::host());
196
197 return {std::move(regions),
198 DataLayout::fromContinuous(continuousFlow.sampleWordSize(), continuousFlow.channelCount(), continuousFlow.channelBufferLength()),
199 continuousFlow.flowInfo()->config.common.maxSyncBatchSizeHint};
200 }
201 else
202 {
203 throw Exception::make(MXL_ERR_UNKNOWN, "Unsupported flow fromat {}", flow.flowInfo()->config.common.format);
204 }
205 }
