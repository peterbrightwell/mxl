346 }
347 count++;
348 }
349 return count;
350 };

372TEST_CASE_PERSISTENT_FIXTURE(mxl::tests::mxlDomainFixture, "DomainWatcher supports concurrent add/remove operations", "[domainwatcher]")
373{
374 // This test spawns multiple threads to add and remove flow watchers concurrently,
375 // to ensure thread safety (no crashes or race conditions and correct final state).
376 auto flowId = *uuids::uuid::from_string("12345678-1234-5678-1234-567812345678");
377 auto mockFiles = MockFlowFiles{domain, flowId};
378
379 std::random_device rd{};
380 std::mt19937 rng{rd()};
381
382 std::atomic<std::size_t> readyCount{0};
383 std::atomic<bool> startSignal{false};
384
385 constexpr auto iterations = 1024;
386 constexpr auto nWorkers = 8;
387 constexpr auto slots = 4;
388
389 struct Worker
390 {
391 std::vector<std::optional<MockWriter>> writers;
392 std::optional<std::thread> thread;
393 };
394
395 auto watcher = std::make_shared<DomainWatcher>(domain);
396 auto work = [&](Worker* w)
397 {
398 readyCount.fetch_add(1);
399 startSignal.wait(false);
400
401 auto dist = std::uniform_int_distribution{std::size_t{0}, w->writers.size() - 1};
402
403 for (std::size_t n = 0; n < iterations; ++n)
404 {
405 auto index = dist(rng);
406 auto& slot = w->writers.at(index);
407 if (slot)
408 {
409 slot.reset();
410 }
411 else
412 {
413 slot.emplace(domain, flowId, watcher);
414 }
415
416 if (index > (w->writers.size() / 2))
417 {
418 mockFiles.touch();
419 }
420 }
421 };
422
423 auto workers = std::list<Worker>{};
424 for (auto i = 0; i < nWorkers; ++i)
425 {
426 auto& w = workers.emplace_back(std::vector<std::optional<MockWriter>>{slots}, std::nullopt);
427 w.thread = std::thread{work, &w};
428 }
429
430 while (readyCount.load() != workers.size())
431 {
432 }
433
434 startSignal.store(true);
435 startSignal.notify_all();
436 // work work work...
437 std::ranges::for_each(workers,
438 [](Worker& w)
439 {
440 w.thread->join();
441 w.writers.clear();
442 });
443
444 REQUIRE(watcher->size() == 0);
445 REQUIRE(watcher->count(flowId) == 0);
446 {
447 auto mockWriter = MockWriter{domain, flowId, watcher};
448 REQUIRE(watcher->size() == 1);
449 REQUIRE(watcher->count(flowId) == 1);
450 }
451 REQUIRE(watcher->size() == 0);
452 REQUIRE(watcher->count(flowId) == 0);
453}
