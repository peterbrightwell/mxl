32 class FILogging
33 {
34 public:
35 void init()
36 {
37 // It is safe to call a logging functions on the libfabric side
38 // while we initialize logging here. We just need to make sure
39 // to initialize logging only once.
40 if (_isInit.exchange(true, std::memory_order_relaxed))
41 {
42 return;
43 }
44
45 auto fiLogLevelCStr = ::getenv("FI_LOG_LEVEL");
46 if (fiLogLevelCStr != nullptr)
47 {
48 auto fiLogLevelStr = std::string{fiLogLevelCStr};
49 std::ranges::transform(fiLogLevelStr, fiLogLevelStr.begin(), [](unsigned char const c) { return std::tolower(c); });
50 auto it = std::ranges::find_if(fiLogLevelStrings,
51 [fiLogLevelStr](std::pair<std::string_view, ::fi_log_level> const& item) { return fiLogLevelStr == item.first; });
52 if (it != fiLogLevelStrings.end())
53 {
54 _level = it->second;
55 }
56 else
57 {
58 _level = FI_LOG_WARN;
59 }
60 }
61
62 auto ops = ::fi_ops_log{
63 .size = sizeof(::fi_ops_log),
64 .enabled = &fiLogEnabled,
65 .ready = &fiLogReady,
66 .log = &fiLog,
67 };
68
69 auto logging = ::fid_logging{
70 .fid = ::fid{},
71 .ops = &ops,
72 };
73
74 fiCall(::fi_import_log, "Failed to initialize logging", fiVersion(), 0, &logging);
75 }
76
77 private:
78 friend int fiLogEnabled(const struct fi_provider*, enum fi_log_level, enum fi_log_subsys, std::uint64_t);

145 void fiLog(const struct fi_provider* prov, enum fi_log_level level, enum fi_log_subsys subsys, char const*, int line, char const* msgIn)
146 {
147 auto msg = std::string{msgIn};
148 if (msg.ends_with('\n'))
149 {
150 msg.pop_back();
151 }
152
153 spdlog::log(
154 {fmt::format("libfabric:{}:{}", fiLogSubsystemName(subsys), prov->name).c_str(), line, ""}, fiLevelToSpdlogLevel(level), "{}", msg);
155 }
