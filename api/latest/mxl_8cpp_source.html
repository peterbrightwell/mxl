44extern "C" MXL_EXPORT
45mxlInstance mxlCreateInstance(char const* in_mxlDomain, char const* in_options)
46{
47 try
48 {
49 auto const opts = (in_options != nullptr) ? in_options : "";
50 auto domainWatcher = std::make_shared<mxl::lib::DomainWatcher>(in_mxlDomain);
51 auto flowIoFactory = std::make_unique<mxl::lib::PosixFlowIoFactory>(domainWatcher);
52 return reinterpret_cast<mxlInstance>(new mxl::lib::Instance{in_mxlDomain, opts, std::move(flowIoFactory), std::move(domainWatcher)});
53 }
54 catch (std::exception& e)
55 {
56 MXL_ERROR("Failed to create instance : {}", e.what());
57 return nullptr;
58 }
59 catch (...)
60 {
61 MXL_ERROR("Failed to create instance");
62 return nullptr;
63 }
64}

66extern "C" MXL_EXPORT
67mxlStatus mxlIsTmpFs(char const* in_path, bool* out_isTmpFs)
68{
69 if ((in_path == nullptr) || (out_isTmpFs == nullptr))
70 {
71 return MXL_ERR_INVALID_ARG;
72 }
73
74#ifdef __linux__
75 using statfs_t = struct statfs;
76 auto buf = statfs_t{};
77 if (::statfs(in_path, &buf) != 0)
78 {
79 return MXL_ERR_UNKNOWN;
80 }
81 *out_isTmpFs = (buf.f_type == TMPFS_MAGIC) || (buf.f_type == RAMFS_MAGIC);
82 return MXL_STATUS_OK;
83#elif defined(__APPLE__)
84 using statfs_t = struct statfs;
85 auto buf = statfs_t{};
86 if (::statfs(in_path, &buf) != 0)
87 {
88 return MXL_ERR_UNKNOWN;
89 }
90 *out_isTmpFs = std::strcmp(buf.f_fstypename, "tmpfs") == 0;
91 return MXL_STATUS_OK;
92#else
93 *out_isTmpFs = false;
94 return MXL_STATUS_OK;
95#endif
96}
