44 inline std::shared_ptr<Domain> getDomain(bool virtualAddress = false, bool rx_cq_data_mode = false)
45 {
46 auto infoList = FabricInfoList::get("127.0.0.1", "9090", Provider::TCP, FI_RMA | FI_WRITE | FI_REMOTE_WRITE, FI_EP_MSG);
47 auto info = *infoList.begin();
48
49 auto fabric = Fabric::open(info);
50 auto domain = Domain::open(fabric);
51
52 if (virtualAddress)
53 {
54 fabric->info()->domain_attr->mr_mode |= FI_MR_VIRT_ADDR;
55 }
56 else
57 {
58 fabric->info()->domain_attr->mr_mode &= ~FI_MR_VIRT_ADDR;
59 }
60
61 if (rx_cq_data_mode)
62 {
63 fabric->info()->rx_attr->mode |= FI_RX_CQ_DATA;
64 }
65 else
66 {
67 fabric->info()->rx_attr->mode &= ~FI_RX_CQ_DATA;
68 }
69
70 return domain;
71 }

78 inline std::pair<MxlRegions, InnerRegions> getHostRegionGroups()
79 {
81 auto innerRegions = std::vector<std::vector<std::uint8_t>>{
82 std::vector<std::uint8_t>(256),
83 std::vector<std::uint8_t>(512),
84 std::vector<std::uint8_t>(1024),
85 std::vector<std::uint8_t>(2048),
86 };
87
88 auto regions = std::vector<Region>{};
89 regions.reserve(innerRegions.size());
90 for (auto const& innerRegion : innerRegions)
91 {
92 regions.emplace_back(*innerRegion.data(), innerRegion.size(), nullptr, nullptr);
93 }
94
95 auto mxlRegions = MxlRegions(regions, DataLayout::fromDiscrete({8, 0, 0, 0}));
96 return {mxlRegions, innerRegions};
97 }
