40 constexpr mxlDataFormat translateFlowFormat(std::string_view format) noexcept
41 {
42 using namespace std::literals;
43
44 constexpr auto const FORMAT_PREFIX = "urn:x-nmos:format:"sv;
45 if (format.starts_with(FORMAT_PREFIX))
46 {
47 auto const tail = format.substr(FORMAT_PREFIX.length());
48 if (tail == "video"sv)
49 {
50 return MXL_DATA_FORMAT_VIDEO;
51 }
52 if (tail == "audio"sv)
53 {
54 return MXL_DATA_FORMAT_AUDIO;
55 }
56 if (tail == "data"sv)
57 {
58 return MXL_DATA_FORMAT_DATA;
59 }
60 }
61
62 return MXL_DATA_FORMAT_UNSPECIFIED;
63 }

99 mxlRational extractRational(picojson::object const& in_object)
100 {
101 auto result = mxlRational{0, 1};
102
103 result.numerator = static_cast<std::int64_t>(fetchAs<double>(in_object, "numerator"));
104 if (auto const it = in_object.find("denominator"); it != in_object.end())
105 {
106 result.denominator = static_cast<std::int64_t>(it->second.get<double>());
107 }
108
109 // Normalize the rational. We should realistically only see x/1 or x/1001 here.
110 auto g = std::gcd(result.numerator, result.denominator);
111 if (g != 0)
112 {
113 result.numerator /= g;
114 result.denominator /= g;
115 }
116
117 return result;
118 }

120 //
121 // Validates that the group hint tag is present and valid
122 // See https://specs.amwa.tv/nmos-parameter-registers/branches/main/tags/grouphint.html
123 //
124 bool validateGroupHint(picojson::object const& in_object)
125 {
126 try
127 {
128 // obtain the tags object. Will throw if not found or not an object
129 auto const tags = fetchAs<picojson::object>(in_object, "tags");
130
131 // obtain the specific tag array from the tags object. Will throw if not found or not an array
132 auto const& groupHints = fetchAs<picojson::array>(tags, "urn:x-nmos:tag:grouphint/v1.0");
133
134 // we need at least a group hint
135 if (groupHints.empty())
136 {
137 MXL_ERROR("Group hint tag found but empty.");
138 return false;
139 }
140 else
141 {
142 // iterate over the array and confirm that all values are strings and follow the
143 // expected format.
144 // "<group-name>:<role-in-group>[:<group-scope>]" where group-scope if present, is either device or node
145 for (auto const& v : groupHints)
146 {
147 if (!v.is<std::string>())
148 {
149 MXL_ERROR("Invalid group hint value. Not a string.");
150 return false;
151 }
152
153 // Get the array item
154 auto const& s = v.get<std::string>();
155 // Split the string into parts separated by ':'
156 auto parts = s | std::views::split(':') |
157 std::views::transform([&](auto&& rng) { return std::string_view(&*rng.begin(), std::ranges::distance(rng)); });
158
159 auto vec = std::vector<std::string_view>{parts.begin(), parts.end()};
160 if ((vec.size() < 2) || (vec.size() > 3))
161 {
162 MXL_ERROR("Invalid group hint value '{}'. Expected format '<group-name>:<role-in-group>[:<group-scope>]'", s);
163 return false;
164 }
165
166 // Validate the group name and role
167 auto const& groupName = vec[0];
168 auto const& role = vec[1];
169 if (groupName.empty() || role.empty())
170 {
171 MXL_ERROR("Invalid group hint value '{}'. Group name and role must not be empty.", s);
172 return false;
173 }
174
175 // Validate the group scope if present
176 if (vec.size() == 3)
177 {
178 auto const& groupScope = vec[2];
179 if (groupScope != "device" && groupScope != "node")
180 {
181 MXL_ERROR("Invalid group hint value '{}'. Group scope must be either 'device' or 'node'.", s);
182 return false;
183 }
184 }
185 }
186 // all the tags passed validation
187 return true;
188 }
189 }
190 catch (std::exception const& e)
191 {
192 MXL_ERROR("Invalid group hint tag. {}", e.what());
193 return false;
194 }

199 FlowParser::FlowParser(std::string const& in_flowDef)
200 : _id{}
201 , _format{MXL_DATA_FORMAT_UNSPECIFIED}
202 , _interlaced{false}
203 , _grainRate{0, 1}
204 , _root{}
205 {
206 //
207 // Parse the json flow
208 //
209 auto jsonValue = picojson::value{};
210 auto const err = picojson::parse(jsonValue, in_flowDef);
211 if (!err.empty())
212 {
213 throw std::invalid_argument{"Invalid JSON flow definition. " + err};
214 }
215
216 // Confirm that the root is a json object
217 if (!jsonValue.is<picojson::object>())
218 {
219 throw std::invalid_argument{"Expected a JSON Object"};
220 }
221 _root = jsonValue.get<picojson::object>();
222
223 auto id = uuids::uuid::from_string(fetchAs<std::string>(_root, "id"));
224 if (!id.has_value())
225 {
226 throw std::invalid_argument{"Invalid flow 'id'."};
227 }
228
229 _id = *id;
230
231 // Read the media format
232 _format = translateFlowFormat(fetchAs<std::string>(_root, "format"));
233
234 // Read the grain rate if this is not an audio flow.
235 if (mxlIsDiscreteDataFormat(_format))
236 {
237 _grainRate = extractRational(fetchAs<picojson::object>(_root, "grain_rate"));
238 }
239 else if (_format == MXL_DATA_FORMAT_AUDIO)
240 {
241 _grainRate = extractRational(fetchAs<picojson::object>(_root, "sample_rate"));
242 }
243 else
244 {
245 throw std::domain_error{"Unsupported flow format."};
246 }
247
248 // Validate that we have a non empty label
249 auto const label = fetchAs<std::string>(_root, "label");
250 if (label.empty())
251 {
252 throw std::domain_error{"Empty flow label."};
253 }
254
255 // Validate that we have a valid group hint tag
256 if (!validateGroupHint(_root))
257 {
258 throw std::domain_error{"Invalid or missing group hint tag."};
259 }
260
261 // Validate that grain_rate if we are interlaced video
262 // Grain rate must be either 30000/1001 or 25/1
263 if (_format == MXL_DATA_FORMAT_VIDEO)
264 {
265 auto const width = static_cast<std::size_t>(fetchAs<double>(_root, "frame_width"));
266 auto const height = static_cast<std::size_t>(fetchAs<double>(_root, "frame_height"));
267
268 if ((width < 2) || (width > MAX_WIDTH) || (height < 1) || (height > MAX_HEIGHT))
269 {
270 auto msg = fmt::format("Invalid video dimensions: {}x{}. range is 2x1 to {}x{}", width, height, MAX_WIDTH, MAX_HEIGHT);
271 throw std::invalid_argument{std::move(msg)};
272 }
273
274 auto interlaceMode = std::string{};
275 if (auto const it = _root.find("interlace_mode"); it != _root.end())
276 {
277 interlaceMode = it->second.get<std::string>();
278 }
279 else
280 {
281 // the interlace_mode field is absent. This means that the flow is progressive. No need to validate the grain rate.
282 // We will use the grain rate as is.
283 interlaceMode = "progressive";
284 }
285
286 constexpr auto validValues = std::array{"progressive", "interlaced_tff", "interlaced_bff"};
287 bool match = std::ranges::any_of(validValues, [&](auto v) { return interlaceMode == v; });
288 if (!match)
289 {
290 auto msg = fmt::format("Invalid interlace_mode: {}", interlaceMode);
291 throw std::invalid_argument{std::move(msg)};
292 }
293
294 if ((interlaceMode == "interlaced_tff") || (interlaceMode == "interlaced_bff"))
295 {
296 // This is an interlaced video flow. confirm that the grain rate is defined to 30000/1001 or 25/1
297 if ((_grainRate != mxlRational{30000, 1001}) && (_grainRate != mxlRational{25, 1}))
298 {
299 auto msg = std::string{"Invalid grain_rate for interlaced video. Expected 30000/1001 or 25/1."};
300 throw std::invalid_argument{std::move(msg)};
301 }
302
303 // The grain rate is valid and we are in interlaced mode. Double the grain rate to express a field rate.
304 _grainRate.numerator *= 2;
305 _interlaced = true;
306 }
307 }

325 std::size_t FlowParser::getPayloadSize() const
326 {
327 auto payloadSize = std::size_t{0};
328
329 if (_format == MXL_DATA_FORMAT_VIDEO)
330 {
331 auto const width = static_cast<std::size_t>(fetchAs<double>(_root, "frame_width"));
332 auto const height = static_cast<std::size_t>(fetchAs<double>(_root, "frame_height"));
333 auto const mediaType = fetchAs<std::string>(_root, "media_type");
334
335 if (mediaType == "video/v210")
336 {
337 if (!_interlaced || ((height % 2) == 0))
338 {
339 // Interlaced media is handled as separate fields.
340 auto const h = _interlaced ? height / 2 : height;
341 payloadSize = getV210LineLength(width) * h;
342 }
343 else
344 {
345 auto msg = std::string{"Invalid video height for interlaced v210. Must be even."};
346 throw std::invalid_argument{std::move(msg)};
347 }
348 }
349 else if (mediaType == "video/v210a")
350 {
351 if (!_interlaced || ((height % 2) == 0))
352 {
353 // Interlaced media is handled as separate fields.
354 auto const h = _interlaced ? height / 2 : height;
355
356 // Fill size
357 auto fillPayloadSize = getV210LineLength(width) * h;
358
359 // Key is stored as 10 bits per pixel, 3x10 bit words in a 32 bit word, little-endian.
360 // the last 2 bits of the 32 bit group of 3 pixels are unused.
361 auto keyPayloadSize = get10BitAlphaLineLength(width) * h;
362
363 // Total payload size is the sum of the fill and key sizes
364 payloadSize = fillPayloadSize + keyPayloadSize;
365 }
366 else
367 {
368 auto msg = std::string{"Invalid video height for interlaced v210a. Must be even."};
369 throw std::invalid_argument{std::move(msg)};
370 }
371 }
372 else
373 {
374 auto msg = std::string{"Unsupported video media_type: "} + mediaType;
375 throw std::invalid_argument{std::move(msg)};
376 }
377 }
378 else if (_format == MXL_DATA_FORMAT_DATA)
379 {
380 auto const mediaType = fetchAs<std::string>(_root, "media_type");
381 if (mediaType == "video/smpte291")
382 {
383 payloadSize = MXL_DATA_FORMAT_GRAIN_SIZE;
384 }
385 else
386 {
387 auto msg = std::string{"Unsupported data media_type: "} + mediaType;
388 throw std::invalid_argument{std::move(msg)};
389 }
390 }
391 else if (_format == MXL_DATA_FORMAT_AUDIO)
392 {
393 // TODO: Also check the media type once we agreed on how to encode
394 // single precision IEEE floats.
395 auto const bitDepth = fetchAs<double>(_root, "bit_depth");
396 if ((bitDepth != 32.0) && (bitDepth != 64.0))
397 {
398 auto msg = fmt::format("Unsupported bit depth: {}", bitDepth);
399 throw std::invalid_argument{std::move(msg)};
400 }
401
402 payloadSize = static_cast<std::size_t>(bitDepth) / 8U;
403 }
404 else
405 {
407 throw std::invalid_argument{"Format not implemented."};
408 }
409
410 return payloadSize;
411 }

413 std::array<std::uint32_t, MXL_MAX_PLANES_PER_GRAIN> FlowParser::getPayloadSliceLengths() const
414 {
415 auto sliceLengths = std::array<std::uint32_t, MXL_MAX_PLANES_PER_GRAIN>{0, 0, 0, 0};
416
417 switch (_format)
418 {
419 case MXL_DATA_FORMAT_DATA:
420 {
421 // For "data" flows, the slice length is 1 byte
422 sliceLengths[0] = 1;
423 return sliceLengths;
424 }
425
426 case MXL_DATA_FORMAT_VIDEO:
427 {
428 // For video flows the slice length is the byte-length of a single
429 // line of v210 video.
430 auto const width = static_cast<std::size_t>(fetchAs<double>(_root, "frame_width"));
431 auto const mediaType = fetchAs<std::string>(_root, "media_type");
432 auto const v210fillSize = getV210LineLength(width);
433
434 if (mediaType == "video/v210")
435 {
436 sliceLengths[0] = v210fillSize;
437 }
438 else if (mediaType == "video/v210a")
439 {
440 sliceLengths[0] = v210fillSize;
441 sliceLengths[1] = get10BitAlphaLineLength(width);
442 }
443 else
444 {
445 auto msg = std::string{"Unsupported video media_type: "} + mediaType;
446 throw std::invalid_argument{std::move(msg)};
447 }
448
449 return sliceLengths;
450 }
451
452 default:
453 {
454 throw std::invalid_argument{"Cannot compute slice length for this data format."};
455 }
456 }

459 std::size_t FlowParser::getTotalPayloadSlices() const
460 {
461 switch (_format)
462 {
463 case MXL_DATA_FORMAT_DATA:
464 {
465 // Since the slice length for data flows is 1 byte, the number of slices must be the
466 // grain size.
467 return MXL_DATA_FORMAT_GRAIN_SIZE;
468 }
469
470 case MXL_DATA_FORMAT_VIDEO:
471 {
472 if (auto const mediaType = fetchAs<std::string>(_root, "media_type"); mediaType != "video/v210" && mediaType != "video/v210a")
473 {
474 auto msg = std::string{"Unsupported video media_type: "} + mediaType;
475 throw std::invalid_argument{std::move(msg)};
476 }
477
478 // For v210, the number of slices is always the number of video lines
479 auto h = static_cast<std::size_t>(fetchAs<double>(_root, "frame_height"));
480 if (_interlaced)
481 {
482 return h / 2;
483 }
484
485 return h;
486 }
487 default:
488 {
489 throw std::invalid_argument{"Cannot compute slice length for this data format."};
490 }
491 }
