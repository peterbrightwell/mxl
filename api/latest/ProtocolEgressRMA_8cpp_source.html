29 SliceRange const& sliceRange, ::fi_addr_t destAddr)
30 {
31 auto const localSize = sliceRange.transferSize(payloadOffset, _layout.sliceSizes[0]);
32 auto const localOffset = sliceRange.transferOffset(payloadOffset, _layout.sliceSizes[0]);
33 auto const remoteSize = sliceRange.transferSize(payloadOffset, _layout.sliceSizes[0]);
34 auto const remoteOffset = sliceRange.transferOffset(payloadOffset, _layout.sliceSizes[0]);
35
36 auto const localRegion = _localRegions[localIndex % _localRegions.size()].sub(localOffset, localSize);
37 auto const remoteRegion = _remoteInfo.remoteRegions[remoteIndex % _remoteInfo.remoteRegions.size()].sub(remoteOffset, remoteSize);
38 auto const remoteSlot = remoteIndex % _remoteInfo.remoteRegions.size();
39
40 _pending += ep.write(_token, localRegion, remoteRegion, destAddr, ImmDataGrain{remoteSlot, sliceRange.end()}.data());
41 }

106 void RMASampleEgressProtocol::registerMemory(std::shared_ptr<Domain> domain)
107 {
108 if (_entryHeaders.empty())
109 {
110 throw Exception::invalidState("Entry headers buffer are not initialized.");
111 }
112
113 auto entryHeaderRegions = std::vector<Region>{};
114 entryHeaderRegions.reserve(_entryHeaders.size());
115 for (auto& header : _entryHeaders)
116 {
117 entryHeaderRegions.emplace_back(reinterpret_cast<std::uintptr_t>(&header),
118 sizeof(AudioEntryHeader),
119 nullptr,
120 nullptr,
121 Region::Location::host()); // Host, because the bounce buffer will always be stored on host memory.
122 }
123
124 // The domain is shared with all endpoints, if there's more than 1 target, we can't assume that the first audio header entry for this target
125 // will be at index 1 (right after the audio samples buffer). Read the number of regions already registered to the domain, then
126 // register the entry header regions, and finally get the local regions again to retrieve the actual regions corresponding to the entry
127 // headers we just registered. Use the offset to correctly get the entry header regions in the list of local regions.
128 auto const offset = domain->localRegions().size();
129 domain->registerRegions(entryHeaderRegions, FI_WRITE);
130 auto const localRegions = domain->localRegions();
131
132 _entryHeaderRegions = std::vector(localRegions.cbegin() + static_cast<std::ptrdiff_t>(offset), localRegions.cend());
133 }

140 void RMASampleEgressProtocol::transferSamples(Endpoint const& ep, std::uint64_t headIndex, std::size_t count, ::fi_addr_t destAddr)
141 {
142 if (count == 0)
143 {
144 throw Exception::invalidArgument("Count must be greater than 0.");
145 }
146
147 auto const entrySizeRequired = (_layout.sampleSize * _layout.channelCount * count) + sizeof(AudioEntryHeader);
148 if (entrySizeRequired > _remoteInfo.bounceBufferInfo->entrySize)
149 {
150 throw Exception::invalidArgument("Count is too large for the bounce buffer entry size. Count {}, entry size {}, required entry size {}.",
151 count,
152 _remoteInfo.bounceBufferInfo->entrySize,
153 entrySizeRequired);
154 }
155
156 // 1- Create the scatter-gather list for the transfer and prepend the audio header.
157 auto sgl = makeScatterGatherList(_layout, headIndex, count, _localRegion);
158 // set the header and prepend it to the scatter-gather list.
159 _entryHeaders[_bounceBufferEntryIndex].headIndex = headIndex;
160 _entryHeaders[_bounceBufferEntryIndex].count = count;
161 sgl.insert(sgl.begin(), _entryHeaderRegions[_bounceBufferEntryIndex]);
162
163 // 2- Get the remote region
164 auto const remoteRegion = _remoteInfo.remoteRegions[_bounceBufferEntryIndex % _remoteInfo.remoteRegions.size()];
165
166 // 3- Send the remote write
167 _pending += ep.write(_token, sgl, remoteRegion, destAddr, _bounceBufferEntryIndex);
168
169 // 4- update bounce buffer entry index for the next transfer
170 _bounceBufferEntryIndex = (_bounceBufferEntryIndex + 1) % _bounceBufferEntryCount;
171 }

188 std::vector<LocalRegion> RMASampleEgressProtocol::makeScatterGatherList(DataLayout::Continuous const& layout, std::uint64_t headIndex,
189 std::size_t count, LocalRegion const& region)
190 {
191 auto slice = mxlMutableWrappedMultiBufferSlice{};
192 AudioBounceBuffer::getMutableMultiBufferSlices(headIndex,
193 count,
194 layout.bufferLength,
195 layout.sampleSize,
196 layout.channelCount,
197 reinterpret_cast<std::uint8_t*>(region.addr), // NOLINT
198 slice);
199
200 // Double the scatter-gather list length if the second fragment is present.
201 auto const sgListLen = (slice.base.fragments[1].size > 0) ? 2 * slice.count : slice.count;
202
203 // Create the scatter-gather list using the slices. We create at least one scatter-gather entry per channel. We potentially create an
204 // additional one per channel if 2 fragments are present (wrap-around). When a fragment is not present its size will be 0.
205 auto sgList = std::vector<LocalRegion>{};
206 sgList.reserve(sgListLen);
207 // Write all fragments for one channel before moving to the next channel.
208 // Doing otherwise (fragment first) only works when source and target wrapped slices split at the same fragment sizes.
209 // If source and target have different ring positions, the target consumes the bytes with different fragment sizes and "stitches" the channel
210 // data back together incorrectly.
211 for (auto chan = std::size_t{0}; chan < slice.count; chan++)
212 {
213 for (auto const& fragment : slice.base.fragments)
214 {
215 // check if the fragment present
216 if (fragment.size > 0)
217 {
218 auto const srcAddr = reinterpret_cast<std::uintptr_t>(fragment.pointer) + (slice.stride * chan);
219
220 sgList.emplace_back(LocalRegion{.addr = srcAddr, .len = fragment.size, .desc = region.desc});
221 }
222 }
223 }
224
225 return sgList;
226 }

233 void RMASampleEgressProtocolTemplate::registerMemory(std::shared_ptr<Domain> domain)
234 {
235 // This function should be called once during setup and be the first memory to register, otherwise it's a bug.
236 if (_localRegion)
237 {
238 throw Exception::invalidState("Memory already registered.");
239 }
240 if (!domain->localRegions().empty())
241 {
242 throw Exception::invalidState("No memory should be previously registered.");
243 }
244
245 // Register the audio region provided by the user. When the actual protocol instance is created, the protocol will register additional
246 // regions for the bounce buffer entry headers, but we can only do that once we have the actual protocol instance since the bounce buffer
247 // entry count is a parameter of the protocol instance.
248 domain->registerRegion(_region, FI_WRITE);
249
250 _localRegion = domain->localRegions().front();
251 }
