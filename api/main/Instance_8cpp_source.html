54 bool parseOptionsJson(std::string const& options, picojson::object& config)
55 {
56 picojson::value parsed;
57 std::string err = picojson::parse(parsed, options);
58 if (!err.empty())
59 {
60 MXL_ERROR("Failed to parse options json: {}", err);
61 return false;
62 }
63
64 if (!parsed.is<picojson::object>())
65 {
66 return false;
67 }
68
69 // Assign the json object config

188 std::tuple<mxlFlowConfigInfo, FlowWriter*, bool> Instance::createFlowWriter(std::string const& flowDef, std::optional<std::string> options)
189 {
190 auto const lock = std::lock_guard{_mutex};
191 auto const parser = FlowParser{flowDef};
192 auto const optionsParser = (options) ? FlowOptionsParser{*options} : FlowOptionsParser{};
193 auto created = false;
194 auto flowData = std::unique_ptr<FlowData>{};
195 FlowWriter* flowWriter = nullptr;
196
197 if (auto const format = parser.getFormat(); mxlIsDiscreteDataFormat(format))
198 {
199 auto [rFlowData, rCreated] = createOrOpenDiscreteFlowData(flowDef, parser, optionsParser);
200 flowData = std::move(rFlowData);
201 created = rCreated;
202 }
203 else if (mxlIsContinuousDataFormat(format))
204 {
205 auto [rFlowData, rWasCreated] = createOrOpenContinuousFlowData(flowDef, parser, optionsParser);
206 flowData = std::move(rFlowData);
207 created = rWasCreated;
208 }
209 else
210 {
211 throw std::runtime_error("Invalid flow format.");
212 }
213
214 auto id = uuids::uuid{flowData->flowInfo()->config.common.id};
215 auto flowConfigInfo = flowData->flowInfo()->config;
216
217 if (auto const pos = _writers.find(id); pos != _writers.end())
218 {
219 auto& v = (*pos).second;
220 v.addReference();
221 flowWriter = v.get();
222 }
223 else
224 {
225 auto writer = _flowIoFactory->createFlowWriter(_flowManager, id, std::move(flowData));
226
227 flowWriter = (*_writers.try_emplace(pos, id, std::move(writer))).second.get();
228 }
229
230 return {flowConfigInfo, flowWriter, created};
231 }

233 std::pair<std::unique_ptr<FlowData>, bool> Instance::createOrOpenDiscreteFlowData(std::string const& flowDef, FlowParser const& parser,
234 FlowOptionsParser const& optionsParser)
235 {
236 // Read the mandatory grain_rate field
237 auto const grainRate = parser.getGrainRate();
238 // Compute the grain count based on our configured history duration
239 auto const grainCount = _historyDuration * grainRate.numerator / (1'000'000'000ULL * grainRate.denominator);
240
241 auto const batchSizeDefault = parser.getTotalPayloadSlices();
242
243 auto [created, flowData] = _flowManager.createOrOpenDiscreteFlow(parser.getId(),
244 flowDef,
245 parser.getFormat(),
246 grainCount,
247 grainRate,
248 parser.getPayloadSize(),
249 parser.getTotalPayloadSlices(),
250 parser.getPayloadSliceLengths(),
251 optionsParser.getMaxSyncBatchSizeHint().value_or(batchSizeDefault),
252 optionsParser.getMaxCommitBatchSizeHint().value_or(batchSizeDefault));
253
254 return {std::move(flowData), created};
255 }

257 std::pair<std::unique_ptr<FlowData>, bool> Instance::createOrOpenContinuousFlowData(std::string const& flowDef, FlowParser const& parser,
258 FlowOptionsParser const& optionsParser)
259 {
260 // Read the mandatory grain_rate field
261 auto const sampleRate = parser.getGrainRate();
262 // Compute the buffer length based on our configured history duration.
263 // The length is divided by 500M instead of 1B to effectively make it twice the
264 // history duration, which is necessary, because only half of the buffer is
265 // accessible for reading at any one point in time.
266 auto const bufferLength = _historyDuration * sampleRate.numerator / (500'000'000ULL * sampleRate.denominator);
267
268 auto const sampleWordSize = parser.getPayloadSize();
269 // FIXME: The page size is just an educated guess to round to for good measure
270 auto const lengthPerPage = 4096U / sampleWordSize;
271
272 auto const pageAlignedLength = ((bufferLength + lengthPerPage - 1U) / lengthPerPage) * lengthPerPage;
273
274 // Default to 10ms worth of samples
275 auto batchSizeDefault = parser.getGrainRate().numerator / (100U * parser.getGrainRate().denominator);
276
277 auto [created, flowData] = _flowManager.createOrOpenContinuousFlow(parser.getId(),
278 flowDef,
279 parser.getFormat(),
280 sampleRate,
281 parser.getChannelCount(),
282 sampleWordSize,
283 pageAlignedLength,
284 optionsParser.getMaxSyncBatchSizeHint().value_or(batchSizeDefault),
285 optionsParser.getMaxCommitBatchSizeHint().value_or(batchSizeDefault));
286
287 return {std::move(flowData), created};
288 }

300 // This function is performed in a 'collaborative best effort' way.
301 // Exceptions thrown should not be propagated to the caller and cause disruptions to the application.
302 // On error the function will return 0 and log the error
303 std::size_t Instance::garbageCollect() const
304 {
305 std::size_t count = 0;
306
307 try
308 {
309 auto base = std::filesystem::path{_flowManager.getDomain()};
310 if (exists(base) && is_directory(base))
311 {
312 for (auto const& entry : std::filesystem::directory_iterator{base})
313 {
314 if (is_directory(entry) && (entry.path().extension() == mxl::lib::FLOW_DIRECTORY_NAME_SUFFIX))
315 {
316 // Try to obtain an exclusive lock on the flow data file. If we can obtain one it means that no
317 // other process is writing to the flow.
318 auto const flowDataFile = mxl::lib::makeFlowDataFilePath(_flowManager.getDomain(), entry.path().stem().string());
319
320 // Check if the flow data file exists
321 if (!std::filesystem::exists(flowDataFile))
322 {
323 MXL_DEBUG("Flow data file {} does not exist", flowDataFile.string());
324 continue;
325 }
326
327 // Open a file descriptor to the flow data file
328 int flags = O_RDONLY | O_CLOEXEC;
329#ifndef __APPLE__
330 flags |= O_NOATIME;
331#endif
332 int fd = ::open(flowDataFile.c_str(), flags);
333 // Try to obtain an exclusive lock on the file descriptor. Do not block if the lock cannot be obtained.
334 bool active = ::flock(fd, LOCK_EX | LOCK_NB) < 0;
335 ::close(fd);
336
337 // The flow is not active. remove it (the folder and everything in it)
338 if (!active)
339 {
340 std::error_code ec;
341 std::filesystem::remove_all(entry.path(), ec);
342 if (ec)
343 {
344 MXL_DEBUG("Failed to remove '{}': {} (error code {})", entry.path().string(), ec.message(), ec.value());
345 }
346 else
347 {
348 count++;
349 }
350 }
351 }
352 }
353 }
354 else
355 {
356 MXL_DEBUG("MXL domain {} does not exist or is not a directory", base.string());
357 }
358 }
359 catch (std::exception const& e)
360 {
361 MXL_DEBUG("Failed to perform garbage collection: {}", e.what());
362 }
363 catch (...)
364 {
365 MXL_DEBUG("Failed to perform garbage collection");
366 }
367 return count;
368 }

370 void Instance::parseOptions(std::string const& options)
371 {
372 // This could be way more sophisticated, but for now we just parse the options and extract the history_duration value if it is present.
373 // A full configuration framework with validation, defaults, etc. could be implemented at a later point.
374
375 // Start with the default history duration
376 std::uint64_t historyDuration = _historyDuration;
377
378 //
379 // Try to parse the options.json file found in the MXL domain directory.
380 // If found and configured, it will override the default history duration.
381 //
382 auto domainOptionsFile = makeDomainOptionsFilePath(_flowManager.getDomain());
383 if (exists(domainOptionsFile))
384 {
385 std::ifstream ifs(domainOptionsFile);
386 if (!ifs)
387 {
388 MXL_ERROR("Failed to open domain options file: {}", domainOptionsFile.string());
389 }
390 else
391 {
392 std::stringstream buffer;
393 buffer << ifs.rdbuf();
394 std::string json_content = buffer.str();
395 picojson::object config;
396 if (parseOptionsJson(json_content, config))
397 {
398 if (auto it = config.find(MXL_HISTORY_DURATION_OPTION); it != config.end() && it->second.is<double>())
399 {
400 MXL_TRACE("Found history duration option in domain specific options: {}ns", it->second.get<double>());
401 historyDuration = static_cast<std::uint64_t>(it->second.get<double>());
402 }
403 }
404 else
405 {
406 MXL_ERROR("Failed to parse domain specific options: {}", options);
407 }
408 }
409 }
410
411 // If we have an instance level options string, parse it as well.
412 if (!options.empty())
413 {
414 picojson::object config;
415 if (parseOptionsJson(options, config))
416 {
417 // We are not considering MXL_HISTORY_DURATION_TAG here. we don't want per-instance history durations.
418 // In the future we might want to use this mecanism to set other options that would be instance specific.
419 }
420 else
421 {
422 MXL_ERROR("Failed to parse instance specific options: {}", options);
423 }
424 }
425
426 // Set the history duration
427 _historyDuration = historyDuration;
428 MXL_DEBUG("History duration set to {} ns", historyDuration);
429 }
