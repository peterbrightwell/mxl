41 void RDMInitiatorTarget::activate(Endpoint& ep)
42 {
43 _state = std::visit(
44 overloaded{
45 [&](Idle) -> State
46 {
47 auto fiAddr = ep.addressVector()->insert(_remoteInfo.fabricAddress);
48 return Activated{.fiAddr = fiAddr};
49 },
50 [](Activated state) -> State { return state; },
51 [](Done) -> State { throw Exception::invalidState("Endpoint has been shutdown and can no longer be used."); },
52 },
53 std::move(_state));
54 }
55
56 void RDMInitiatorTarget::shutdown(Endpoint& ep)
57 {
58 _state = std::visit(
59 overloaded{
60 [](Idle) -> State
61 {
62 MXL_WARN("Shutdown requested while waiting to activate, aborting.");
63 return Done{};
64 },
65 [&](Activated state) -> State
66 {
67 ep.addressVector()->remove(state.fiAddr);
68 return Done{};
69 },
70 [](Done state) -> State { return state; },
71 },
72 std::move(_state));
73 }

103 void RDMInitiatorTarget::handleCompletion(Endpoint&, Completion completion)
104 {
105 if (completion.isErrEntry())
106 {
107 MXL_ERROR("Completion error.");
108 return;
109 }
110
111 std::visit(
112 overloaded{
113 [](Idle const&) {},
114 [&](Activated const&) { _proto->processCompletion(completion.data()); },
115 [](Done const&) {},

120 std::unique_ptr<RDMInitiator> RDMInitiator::setup(mxlFabricsInitiatorConfig const& config)
121 {
122 auto provider = providerFromAPI(config.interface.provider);
123 if (!provider)
124 {
125 throw Exception::make(MXL_ERR_NO_FABRIC, "No provider available.");
126 }
127
128 auto caps = FI_RMA | FI_WRITE;
129 // To enable device memory support:
130 // caps |= FI_HMEM;
131 auto fabricInfoList = FabricInfoList::get(config.interface.address.node, config.interface.address.service, provider.value(), caps, FI_EP_RDM);
132 if (fabricInfoList.begin() == fabricInfoList.end())
133 {
134 throw Exception::make(MXL_ERR_NO_FABRIC, "No suitable fabric available");
135 }
136
137 auto info = *fabricInfoList.begin();
138 MXL_DEBUG("{}", fi_tostr(info.raw(), FI_TYPE_INFO));
139
140 auto fabric = Fabric::open(info);
141 auto domain = Domain::open(fabric);
142
143 auto endpoint = Endpoint::create(domain);
144
145 auto cqAttr = CompletionQueue::Attributes::defaults();
146 if (provider == Provider::EFA)
147 {
148 if (!CompletionQueue::isWaitObjectSupportedForEFA())
149 {
150 MXL_WARN("Wait objects not supported in EFA provider for this libfabric version. Only non-blocking API available.");
151 cqAttr.waitObject = FI_WAIT_NONE;
152 }
153 }
154 auto cq = CompletionQueue::open(endpoint.domain(), cqAttr);
155 endpoint.bind(cq, FI_TRANSMIT | FI_RECV);
156
157 auto av = AddressVector::open(endpoint.domain());
158 endpoint.bind(av);
159
160 endpoint.enable();
161
162 auto regions = MxlRegions::forReader(config.reader);
163 auto proto = selectEgressProtocol(regions.dataLayout(), regions.regions());
164
165 proto->registerMemory(domain);

182 void RDMInitiator::addTarget(TargetInfo const& targetInfo)
183 {
184 if (_remoteEndpoints.contains(targetInfo.id))
185 {
186 throw Exception::exists("A target with endpoint id {} has already been added to this initiator.", targetInfo.id);
187 }
188
189 auto const token = Completion::randomToken();
190 auto proto = _proto->createInstance(token, targetInfo);
191 proto->registerMemory(_endpoint.domain());
192
193 _remoteEndpoints.emplace(targetInfo.id, token);
194 _targets.emplace(token, RDMInitiatorTarget(std::move(proto), targetInfo));
195 }

207 void RDMInitiator::transferGrain(std::uint64_t grainIndex, std::uint16_t startSlice, std::uint16_t endSlice)
208 {
209 // Post a transfer work item to all targets. If the target is not in "Added" state
210 // this is a no-op.
211 for (auto& [_, target] : _targets)
212 {
213 // A completion will be posted to the completion queue, after which the counter will be decremented again
214 target.transferGrain(_endpoint, grainIndex, grainIndex, MXL_GRAIN_PAYLOAD_OFFSET, SliceRange::make(startSlice, endSlice));
215 }
216 }

241 // makeProgressBlocking
242 bool RDMInitiator::makeProgressBlocking(std::chrono::steady_clock::duration timeout)
243 {
244 auto now = std::chrono::steady_clock::now();
245 activateIdleEndpoints();
246 auto elapsed = std::chrono::steady_clock::now() - now;
247
248 auto remaining = timeout - elapsed;
249 if (remaining.count() >= 0)
250 {
251 blockOnCQ(remaining);
252 }
253 else
254 {
255 pollCQ();
256 }
257
258 return hasPendingWork();
259 }

296 void RDMInitiator::blockOnCQ(std::chrono::steady_clock::duration timeout)
297 {
298 // A zero timeout would cause the queue to block indefinetly, which
299 // is not our documented behaviour.
300 if (timeout == std::chrono::milliseconds::zero())
301 {
302 // So just behave exactly like the non-blocking variant.
303 pollCQ();
304 return;
305 }
306
307 if (auto completion = _endpoint.completionQueue()->readBlocking(timeout); completion)
308 {
309 processCompletion(*completion);
310 }
311 }
312
313 void RDMInitiator::pollCQ()
314 {
315 if (auto completion = _endpoint.completionQueue()->read(); completion)
316 {
317 processCompletion(*completion);
318 }
319 }

329 void RDMInitiator::processCompletion(Completion completion)
330 {
331 auto it = _targets.find(completion.token());
332 if (it == _targets.end())
333 {
334 MXL_ERROR("Dropping completion for unknown target.");
335 return;
336 }
337
338 it->second.handleCompletion(_endpoint, completion);
339 }
