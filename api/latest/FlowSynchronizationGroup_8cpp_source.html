32 void FlowSynchronizationGroup::addReader(DiscreteFlowReader const& reader, std::uint16_t minValidSlices)
33 {
34 auto prev = _readers.before_begin();
35 for (auto current = std::next(prev); current != _readers.end(); ++current)
36 {
37 if (current->reader == &reader)
38 {

77 mxlStatus FlowSynchronizationGroup::waitForDataAt(Timepoint originTime, Timepoint deadline) const
78 {
79 for (auto it = _readers.begin(); it != _readers.end(); /* Nothing */)
80 {
81 auto const current = it++;
82
83 auto const expectedIndex = timestampToIndex(current->grainRate, originTime);
84 auto const runtimeInfo = current->reader->getFlowRuntimeInfo();
85 if (expectedIndex > runtimeInfo.headIndex)
86 {
87 auto result = mxlStatus{MXL_ERR_UNKNOWN};
88 switch (current->variant)
89 {
90 case Variant::Discrete:
91 result =
92 static_cast<DiscreteFlowReader const*>(current->reader)->waitForGrain(expectedIndex, current->minValidSlices, deadline);
93 break;
94
95 case Variant::Continuous:
96 result = static_cast<ContinuousFlowReader const*>(current->reader)->waitForSamples(expectedIndex, deadline);
97 break;
98 }
99
100 if (result == MXL_STATUS_OK)
101 {
102 // If the current source delay of this flow exceeds any previosuly observed source delay of
103 // this flow we update the cached maximum and if this new maximum turns out to be bigger than
104 // the maximum source delay observed for the flow at the head of the list, we move this flow
105 // to the front, hoping that we can save blocking waits in the future.
106 auto const expectedArrivalTime = indexToTimestamp(current->grainRate, expectedIndex);
107 auto const currentTaiTime = currentTime(Clock::TAI);
108 if (currentTaiTime > expectedArrivalTime)
109 {
110 auto const sourceDelay = (currentTaiTime - expectedArrivalTime).value;
111 if (sourceDelay > current->maxObservedSourceDelay)
112 {
113 current->maxObservedSourceDelay = sourceDelay;
114 if (current->maxObservedSourceDelay > _readers.begin()->maxObservedSourceDelay)
115 {
116 _readers.splice_after(_readers.before_begin(), _readers, current);
117 }
118 }
119 }
120 }
121 else
122 {
123 return result;
124 }
125 }
126 }
127 return MXL_STATUS_OK;
128 }
