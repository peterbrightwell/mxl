20 std::pair<std::unique_ptr<RDMTarget>, std::unique_ptr<TargetInfo>> RDMTarget::setup(mxlFabricsTargetConfig const& config,
21 TargetSetupOptions const& options)
22 {
23 // Both fields may arrive as null at this call site — libfabric's FI_SOURCE path
24 // accepts that as "bind to any local address". fmt's `{}` formatter for char const*
25 // throws fmt::format_error("string pointer is null") on a nullptr, which MXL_INFO
26 // silently swallows. Substitute a printable sentinel for the log only — the original
27 // pointers still go to fi_getinfo so the wildcard semantics are preserved.
28 char const* const nodeStr = config.interface.address.node ? config.interface.address.node : "<unspecified>";
29 char const* const serviceStr = config.interface.address.service ? config.interface.address.service : "<unspecified>";
30
31 MXL_INFO("setting up target [endpoint = {}:{}, provider = {}]", nodeStr, serviceStr, config.interface.provider);
32
33 auto provider = providerFromAPI(config.interface.provider);
34 if (!provider)
35 {
36 throw Exception::invalidArgument("Invalid provider specified");
37 }
38
39 std::uint64_t caps = FI_RMA | FI_REMOTE_WRITE;
40 // To enable device memory support:
41 // caps |= FI_HMEM;
42 auto fabricInfoList = FabricInfoList::get(config.interface.address.node, config.interface.address.service, provider.value(), caps, FI_EP_RDM);
43 if (fabricInfoList.begin() == fabricInfoList.end())
44 {
45 throw Exception::make(MXL_ERR_NO_FABRIC, "No suitable fabric available");
46 }
47
48 auto info = *fabricInfoList.begin();
49 MXL_DEBUG("{}", fi_tostr(info.raw(), FI_TYPE_INFO));
50
51 auto fabric = Fabric::open(info);
52 auto domain = Domain::open(fabric);
53
54 auto endpoint = Endpoint::create(domain);
55
56 auto cqAttr = CompletionQueue::Attributes::defaults();
57 if (options.cqDepth)
58 {
59 cqAttr.size = *options.cqDepth;
60 }
61 if (provider == Provider::EFA)
62 {
63 if (!CompletionQueue::isWaitObjectSupportedForEFA())
64 {
65 MXL_WARN("Wait objects not supported in EFA provider for this libfabric version. Only non-blocking API available.");
66 cqAttr.waitObject = FI_WAIT_NONE;
67 }
68 }
69 auto cq = CompletionQueue::open(domain, cqAttr);
70 endpoint.bind(cq, FI_RECV | FI_TRANSMIT);
71
72 // Connectionless endpoints must be bound to an address vector. Even if it is not using the address vector.
73 auto av = AddressVector::open(domain);
74 endpoint.bind(av);
75
76 // Connectionless endpoints must be explictely enabled when they are ready to be used.
77 endpoint.enable();
78
79 auto mxlRegions = MxlRegions::forWriter(config.writer);
80 auto protocol = selectIngressProtocol(mxlRegions.dataLayout(), mxlRegions.regions(), mxlRegions.maxSyncBatchSize());
81 auto targetInfo = std::make_unique<TargetInfo>(
82 endpoint.id(), endpoint.localAddress(), protocol->registerMemory(domain), protocol->bounceBufferInfo());

100 {

103 throw Exception::unsupportedOperation("The current protocol does not support reading grains.");

105
106 if (auto res = readNext<QueueReadMode::NonBlocking>({}); res)
107 {

158 template<QueueReadMode queueReadMode>
159 std::optional<Target::ReadResult> RDMTarget::readNext(std::chrono::steady_clock::duration timeout)
160 {
161 auto completion = readCompletionQueue<queueReadMode>(*_ep.completionQueue(), timeout);
162 if (completion)
163 {
164 return _protocol->read(_ep, *completion);
165 }
166
167 return {};
168 }
