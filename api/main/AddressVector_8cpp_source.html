54 fi_addr_t AddressVector::insert(FabricAddress const& addr)
55 {
56 ::fi_addr_t fiAddr{FI_ADDR_UNSPEC};
57
58 if (auto ret = ::fi_av_insert(_raw, addr.raw(), 1, &fiAddr, 0, nullptr); ret != 1)
59 {
60 throw Exception::internal("Failed to insert address into the address vector. {}", ::fi_strerror(ret));
61 }
62 MXL_INFO("Remote endpoint address \"{}\" was added to the Address Vector with fi_addr \"{}\"", addrToString(addr), fiAddr);
63
64 return fiAddr;
65 }

72 std::string AddressVector::addrToString(FabricAddress const& addr) const
73 {
74 std::string s;
75 std::size_t len = 0;
76
77 ::fi_av_straddr(_raw, addr.raw(), nullptr, &len);
78 s.resize(len);
79
80 auto ret = ::fi_av_straddr(_raw, addr.raw(), s.data(), &len);
81 if (ret == nullptr)
82 {
83 throw Exception::internal("Failed to convert address to string");
84 }
85
86 return s;
87 }
