122 std::optional<Completion> CompletionQueue::handleReadResult(ssize_t ret, ::fi_cq_data_entry const& entry)
123 {
124 if (ret == -FI_EAGAIN)
125 {
126 // No entry available
127 return std::nullopt;
128 }
129
130 if (ret == -FI_EAVAIL)
131 {
132 // An entry is available but in the error queue
133 ::fi_cq_err_entry err;
134 fi_cq_readerr(_raw, &err, 0);
135
136 return Completion{
137 Completion::Error{err, this->shared_from_this()}
138 };
139 }
140
141 if (ret < 0)
142 {
143 throw FabricException::make(ret, "Failed to read completion from queue: {}", ::fi_strerror(ret));
144 }
145
146 return Completion{Completion::Data{entry}};
147 }
