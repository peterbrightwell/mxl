379TEST_CASE_PERSISTENT_FIXTURE(mxl::tests::mxlDomainFixture, "DomainWatcher supports concurrent add/remove operations", "[domainwatcher]")
380{
381 // This test spawns multiple threads to add and remove flow watchers concurrently,
382 // to ensure thread safety (no crashes or race conditions and correct final state).
383 auto flowId = *uuids::uuid::from_string("12345678-1234-5678-1234-567812345678");
384 auto mockFiles = MockFlowFiles{domain, flowId};
385
386 std::random_device rd{};
387 std::mt19937 rng{rd()};
388
389 std::atomic<std::size_t> readyCount{0};
390 std::atomic<bool> startSignal{false};
391
392 constexpr auto iterations = 1024;
393 constexpr auto nWorkers = 8;
394 constexpr auto slots = 4;
395
396 struct Worker
397 {
398 std::vector<std::optional<MockWriter>> writers;
399 std::optional<std::thread> thread;
400 };
401
402 auto watcher = std::make_shared<DomainWatcher>(domain);
403 auto work = [&](Worker* w)
404 {
405 readyCount.fetch_add(1);
406 startSignal.wait(false);
407
408 auto dist = std::uniform_int_distribution{std::size_t{0}, w->writers.size() - 1};
409
410 for (std::size_t n = 0; n < iterations; ++n)
411 {
412 auto index = dist(rng);
413 auto& slot = w->writers.at(index);
414 if (slot)
415 {
416 slot.reset();
417 }
418 else
419 {
420 slot.emplace(domain, flowId, watcher);
421 }
422
423 if (index > (w->writers.size() / 2))
424 {
425 mockFiles.touch();
426 }
427 }
428 };
429
430 auto workers = std::list<Worker>{};
431 for (auto i = 0; i < nWorkers; ++i)
432 {
433 auto& w = workers.emplace_back(std::vector<std::optional<MockWriter>>{slots}, std::nullopt);
434 w.thread = std::thread{work, &w};
435 }
436
437 while (readyCount.load() != workers.size())
438 {
439 }
440
441 startSignal.store(true);
442 startSignal.notify_all();
443 // work work work...
444 std::ranges::for_each(workers,
445 [](Worker& w)
446 {
447 w.thread->join();
448 w.writers.clear();
449 });
450
451 REQUIRE(watcher->size() == 0);
452 REQUIRE(watcher->count(flowId) == 0);
453 {
454 auto mockWriter = MockWriter{domain, flowId, watcher};
455 REQUIRE(watcher->size() == 1);
456 REQUIRE(watcher->count(flowId) == 1);
457 }
458 REQUIRE(watcher->size() == 0);
459 REQUIRE(watcher->count(flowId) == 0);
460}
