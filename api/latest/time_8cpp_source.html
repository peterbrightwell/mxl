49MXL_EXPORT
50uint64_t mxlGetNsUntilIndex(uint64_t index, mxlRational const* editRate)
51{
52 // Validate the edit rate
53 if ((editRate == nullptr) || (editRate->denominator == 0) || (editRate->numerator == 0))
54 {
55 return MXL_UNDEFINED_INDEX;
56 }
57
58 auto const targetNs = mxlIndexToTimestamp(editRate, index);
59 auto const nowNs = mxlGetTime();
60 return ((nowNs != 0ULL) && (targetNs >= nowNs)) ? targetNs - nowNs : 0ULL;
61}
