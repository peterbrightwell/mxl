45 inline std::shared_ptr<Domain> getDomain(bool virtualAddress = false, bool rx_cq_data_mode = false)
46 {
47 auto infoList = FabricInfoList::get("127.0.0.1", "9090", Provider::TCP, FI_RMA | FI_WRITE | FI_REMOTE_WRITE, FI_EP_MSG);
48 auto info = *infoList.begin();
49
50 auto fabric = Fabric::open(info);
51 auto domain = Domain::open(fabric);
52
53 if (virtualAddress)
54 {
55 fabric->info()->domain_attr->mr_mode |= FI_MR_VIRT_ADDR;
56 }
57 else
58 {
59 fabric->info()->domain_attr->mr_mode &= ~FI_MR_VIRT_ADDR;
60 }
61
62 if (rx_cq_data_mode)
63 {
64 fabric->info()->rx_attr->mode |= FI_RX_CQ_DATA;
65 }
66 else
67 {
68 fabric->info()->rx_attr->mode &= ~FI_RX_CQ_DATA;
69 }
70
71 return domain;
72 }
