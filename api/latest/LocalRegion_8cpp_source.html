56 LocalRegionGroupSpan LocalRegionGroup::span(std::size_t begin, std::size_t end) const
57 {
58 if (end < begin)
59 {
60 throw Exception::invalidArgument("end {} is smaller than begin {}", end, begin);
61 }
62
63 auto const spanLength = end - begin;
64 if (spanLength > _inner.size())
65 {
66 throw Exception::invalidArgument(
67 "requested span size {} will be bigger than the actual size of the full vector {}", spanLength, _inner.size());
68 }
69
70 return LocalRegionGroupSpan{
71 std::span{_inner.cbegin() + begin, spanLength},
72 std::span{_iovs.cbegin() + begin, spanLength},
73 std::span{_descs.cbegin() + begin, spanLength},
74 };
75 }
