59 inline Grain* DiscreteFlowData::emplaceGrain(char const* grainFilePath, std::size_t grainPayloadSize)
60 {
61 auto const mode = this->created() ? AccessMode::CREATE_READ_WRITE : this->accessMode();
62 auto grain = SharedMemoryInstance<Grain>{grainFilePath, mode, grainPayloadSize, LockMode::Shared};
63
64 if (!this->created())
65 {
66 // Check for the version of the grain data structure in the memory that was just mapped.
67 if (grain.get()->header.info.version != GRAIN_HEADER_VERSION)
68 {
69 throw std::invalid_argument{
70 fmt::format("Unsupported grain version: {}, supported version is: {}", grain.get()->header.info.version, GRAIN_HEADER_VERSION)};
71 }
72 }
73
74 return _grains.emplace_back(std::move(grain)).get();
75 }
