127 MxlRegions mxlRegionsFromFlow(FlowData& flow)
128 {
129 static_assert(sizeof(GrainHeader) == 8192,
130 "GrainHeader type size changed! The Fabrics API makes assumptions on the memory layout of a flow, please review the code below if the "
131 "change is intended!");
132
133 if (mxlIsDiscreteDataFormat(flow.flowInfo()->config.common.format))
134 {
135 auto& discreteFlow = static_cast<DiscreteFlowData&>(flow);
136 std::vector<Region> regions;
137
138 for (std::size_t i = 0; i < discreteFlow.grainCount(); ++i)
139 {
140 auto grain = discreteFlow.grainAt(i);
141
142 auto grainInfoBaseAddr = reinterpret_cast<std::uintptr_t>(discreteFlow.grainAt(i));
143 auto grainInfoSize = sizeof(GrainHeader);
144 auto grainPayloadSize = grain->header.info.grainSize;
145
146 if (flow.flowInfo()->config.common.payloadLocation != MXL_PAYLOAD_LOCATION_HOST_MEMORY)
147 {
148 throw Exception::make(MXL_ERR_UNKNOWN,
149 "GPU memory is not currently supported in the Flow API of MXL. Edit the code below when it is supported");
150 }
151
152 regions.emplace_back(grainInfoBaseAddr, grainInfoSize + grainPayloadSize, Region::Location::host());
153 }
154
155 return {std::move(regions),
156 /*DataLayout::fromVideo(false)*/};
157 }
158 else if (mxlIsContinuousDataFormat(flow.flowInfo()->config.common.format))
159 {
160 auto& continuousFlow = static_cast<ContinuousFlowData&>(flow);
161 std::vector<Region> regions;
162
163 // For the continuous flow, the data layout is a single contiguous buffer
164 regions.emplace_back(
165 reinterpret_cast<std::uintptr_t>(continuousFlow.channelData()), continuousFlow.channelDataLength(), Region::Location::host());
166
167 return {
168 std::move(regions),
169 /*DataLayout::fromAudio(continuousFlow.channelCount(),
170 continuousFlow.channelBufferLength(),
171 continuousFlow.sampleWordSize()),*/
172 };
173 }
174
175 else
176 {
177 throw Exception::make(MXL_ERR_UNKNOWN, "Unsupported flow fromat {}", flow.flowInfo()->config.common.format);
178 }
179 }

181 MxlRegions mxlRegionsFromUser(mxlFabricsMemoryRegion const* regions, size_t count)
182 {
183 std::vector<Region> outRegions;
184 for (size_t i = 0; i < count; i++)
185 {
186 outRegions.emplace_back(regions[i].addr, regions[i].size, Region::Location::fromAPI(regions[i].loc));
187 }
188
189 return {std::move(outRegions) /*, DataLayout::fromVideo(false)*/}; // TODO: datalayout struct definition at API level
