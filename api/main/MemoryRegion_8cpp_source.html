38 MemoryRegion MemoryRegion::reg(Domain& domain, Region const& region, std::uint64_t access)
39 {
40 ::fid_mr* raw;
41 std::random_device rd;
42 std::mt19937_64 gen(rd());
43 std::uniform_int_distribution<uint64_t> dist(0, UINT64_MAX);
44
45 std::uint64_t flags = 0;
46 flags |= region.loc.isHost() ? 0 : FI_HMEM_DEVICE_ONLY;
47
48 ::fi_mr_attr attr{};
49 setDeviceId(attr, region.loc);
50
51 attr.mr_iov = region.asIovec();
52 attr.iov_count = 1;
53 attr.access = access;
54 attr.offset = 0; // reserved to 0
55 attr.requested_key = dist(gen); // creating a random key, but it can be ignored if FI_MR_PROV_KEY mr mode is set
56 attr.context = nullptr; // not used
57 attr.auth_key_size = 0;
58 attr.auth_key = nullptr;
59 attr.iface = region.loc.iface();
60 attr.hmem_data = nullptr;
61 attr.page_size = 4096; // not used'
62 attr.base_mr = nullptr; // not used
63 attr.sub_mr_cnt = 0; // not used
64
65 fiCall(fi_mr_regattr, "Failed to register memory region", domain.raw(), &attr, flags, &raw);
66
67 MXL_DEBUG("Registered memory region with address 0x{}, size {} and location {} -> key 0x{:x} desc {}",
68 region.base,
69 region.size,
70 region.loc.toString(),
71 raw->key,
72 raw->mem_desc);
