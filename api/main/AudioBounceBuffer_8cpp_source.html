15 void AudioBounceBuffer::getMutableMultiBufferSlices(std::uint64_t index, std::size_t count, std::size_t bufferLength, std::size_t sampleWordSize,
16 std::size_t channelCount, std::uint8_t* baseBufferPtr, mxlMutableWrappedMultiBufferSlice& slice) noexcept
17 {
18 auto const startOffset = (index + bufferLength - count) % bufferLength;
19 auto const endOffset = (index % bufferLength);
20
21 auto const firstLength = (startOffset < endOffset) ? count : bufferLength - startOffset;
22 auto const secondLength = count - firstLength;
23
24 slice.base.fragments[0].pointer = baseBufferPtr + (sampleWordSize * startOffset);
25 slice.base.fragments[0].size = sampleWordSize * firstLength;
26
27 slice.base.fragments[1].pointer = baseBufferPtr;
28 slice.base.fragments[1].size = sampleWordSize * secondLength;
29
30 slice.stride = sampleWordSize * bufferLength;
31 slice.count = channelCount;
32 }

85 AudioEntryHeader const& AudioBounceBuffer::unpack(std::size_t entryIndex, Region const& outRegion) const
86 {
87 auto const& entry = _entries.at(entryIndex);
88 auto const* header = entry.header();
89
90 auto const maxCountPerEntry = (entrySize() - sizeof(AudioEntryHeader)) / (_layout.channelCount * _layout.sampleSize);
91
92 if (header->count > maxCountPerEntry)
93 {
94 throw Exception::invalidArgument(
95 "Invalid 'count' {} received in the header. That number of samples per channel would bust the bounce buffer entry.", header->count);
96 }
97
98 // Using the given audio data layout, head index and number of samples recover the slices
99 mxlMutableWrappedMultiBufferSlice slices;
100 getMutableMultiBufferSlices(header->headIndex,
101 header->count,
102 _layout.bufferLength,
103 _layout.sampleSize,
104 _layout.channelCount,
105 reinterpret_cast<std::uint8_t*>(outRegion.base), // NOLINT
106 slices);
107
108 auto srcAddr = entry.samples();
109 // The egress side writes the payload in channel-major order: all fragments for one channel are contiguous before the next channel.
110 // Consume fragments in that same order so wrapped slices do not interleave channel data.
111 for (auto chan = std::size_t{0}; chan < slices.count; chan++)
112 {
113 for (auto const& fragment : slices.base.fragments)
114 {
115 // check if the fragment present
116 if (fragment.size > 0)
117 {
118 auto const dstAddr = reinterpret_cast<std::uint8_t*>(fragment.pointer) + (slices.stride * chan);
119 std::memcpy(dstAddr, srcAddr, fragment.size);
120 srcAddr += fragment.size;
121 }
122 }
123 }
124
125 return *header;
126 }
