54 mxlStatus PosixContinuousFlowReader::waitForSamples(std::uint64_t index, Timepoint deadline) const
55 {
56 if (_flowData)
57 {
58 auto const result = getSamplesImpl(index, 0U, deadline, nullptr);
59
60 // If we were ultimately too early, even with blocking for a
61 // certain amount of time it could very well be that we're
62 // operating on a stale flow, so we use the opportunity to check
63 // whether it's valid.
64 return ((result != MXL_ERR_OUT_OF_RANGE_TOO_EARLY) || isFlowValidImpl()) ? result : MXL_ERR_FLOW_INVALID;
65 }
66
67 return MXL_ERR_UNKNOWN;
68 }

112 struct stat st;

120 mxlStatus PosixContinuousFlowReader::getSamplesImpl(std::uint64_t index, std::size_t count,
121 mxlWrappedMultiBufferSlice* payloadBuffersSlices) const
122 {
123 if (auto const headIndex = _flowData->flowInfo()->runtime.headIndex; index <= headIndex)
124 {
125 auto const minIndex = (headIndex >= (_bufferLength / 2U)) ? (headIndex - (_bufferLength / 2U)) : std::uint64_t{0};
126
127 if ((index >= minIndex) && ((index - minIndex) >= count))
128 {
129 if (payloadBuffersSlices != nullptr)
130 {
131 auto const startOffset = (index + _bufferLength - count) % _bufferLength;
132 auto const endOffset = (index % _bufferLength);
133
134 auto const firstLength = (startOffset < endOffset) ? count : _bufferLength - startOffset;
135 auto const secondLength = count - firstLength;
136
137 auto const baseBufferPtr = static_cast<std::uint8_t const*>(_flowData->channelData());
138 auto const sampleWordSize = _flowData->sampleWordSize();
139
140 payloadBuffersSlices->base.fragments[0].pointer = baseBufferPtr + sampleWordSize * startOffset;
141 payloadBuffersSlices->base.fragments[0].size = sampleWordSize * firstLength;
142
143 payloadBuffersSlices->base.fragments[1].pointer = baseBufferPtr;
144 payloadBuffersSlices->base.fragments[1].size = sampleWordSize * secondLength;
145
146 payloadBuffersSlices->stride = sampleWordSize * _bufferLength;
147 payloadBuffersSlices->count = _channelCount;
148 }
149
150 return MXL_STATUS_OK;
151 }
152
153 return MXL_ERR_OUT_OF_RANGE_TOO_LATE;
154 }
155
156 return MXL_ERR_OUT_OF_RANGE_TOO_EARLY;
157 }

159 mxlStatus PosixContinuousFlowReader::getSamplesImpl(std::uint64_t index, std::size_t count, Timepoint deadline,
160 mxlWrappedMultiBufferSlice* payloadBuffersSlices) const
161 {
162 auto const flow = _flowData->flow();
163 auto const syncObject = std::atomic_ref{flow->state.syncCounter};
164 while (true)
165 {
166 auto const previousSyncCounter = syncObject.load(std::memory_order_acquire);
167 auto const result = getSamplesImpl(index, count, payloadBuffersSlices);
168 // NOTE: Before C++26 there is no way to access the address of the object wrapped
169 // by an atomic_ref. If there were it would be much more appropriate to pass
170 // syncObject by reference here and only unwrap the underlying integer in the
171 // implementation of waitUntilChanged.
172 if ((result != MXL_ERR_OUT_OF_RANGE_TOO_EARLY) || !waitUntilChanged(&flow->state.syncCounter, previousSyncCounter, deadline))
173 {
174 return result;
175 }
176 }
177 }
