240 std::pair<std::optional<Completion>, std::optional<Event>> Endpoint::readQueuesBlocking(std::chrono::steady_clock::duration timeout)
241 {
242 std::optional<Completion> completion{std::nullopt};
243 std::optional<Event> event{std::nullopt};
244
245 // No queues, no blocking;
246 if (!_eq && !_cq)
247 {
248 throw Exception::invalidState("No queues bound to endpoint");
249 }
250
251 // There is no event queue, so we just block on the completion queue.
252 if (!_eq)
253 {
254 return {(*_cq)->readBlocking(timeout), event};
255 }
256
257 auto timeoutMs = std::chrono::duration_cast<std::remove_const_t<decltype(EQReadInterval)>>(timeout);
258
259 // This is the simple case: The user-supplied timeout is less than the minimum EventQueue read interval. So we just read the EventQueue once
260 // and then block on the CompletionQueue until the specified timeout.
261 if (timeoutMs < EQReadInterval)
262 {
263 auto event = (*_eq)->read();
264 auto completion = (*_cq)->readBlocking(timeout);
265
266 return {completion, event};
267 }
268
269 // The time point at which we must return control to the user.
270 auto deadline = std::chrono::steady_clock::now() + timeout;
271
272 for (;;)
273 {
274 // The maximum duration we can block for until we must return control to the caller.
275 auto timeUntilDeadline = std::chrono::duration_cast<std::chrono::milliseconds>(deadline - std::chrono::steady_clock::now());
276
277 // If the time is up, we return control.
278 if (timeUntilDeadline <= std::chrono::milliseconds(0))
279 {
280 return {completion, event};
281 }
282
283 // Do a non-blocking read of the event queue first.
284 event = (*_eq)->read();
285 if (event)
286 {
287 return {completion, event};
288 }
289
290 // And then block on the completion queue until either the minimum duration between event queue reads has elapsed, or the latest point in
291 // time at which we need to return control to the caller has been reached.
292 completion = (*_cq)->readBlocking(std::min(EQReadInterval, timeUntilDeadline));
293 if (completion)
294 {
295 return {completion, event};
296 }
297 }
298 }

354 std::size_t Endpoint::write(Completion::Token token, LocalRegionGroup const& localGroup, RemoteRegion const& remote, ::fi_addr_t destAddr,
355 std::optional<std::uint32_t> immData) const
356 {
357 auto remoteRmaIov = remote.toRmaIov();
358 auto remoteOffset = std::size_t{0};
359
360 auto const iovLimit = info().txIovLimit();
361 auto const nbWrites = (localGroup.size() + iovLimit - 1) / iovLimit; // ceil(group.size() / iovLimit)
362
363 for (auto i = std::size_t{0}; i < nbWrites; i++)
364 {
365 auto const begin = i * iovLimit;
366 auto const end = begin + std::min(iovLimit, localGroup.size() - begin);
367
368 auto const localGroupSpan = localGroup.span(begin, end);
369
370 // Validate that we don't bust the remote region
371 if ((remoteOffset + localGroupSpan.byteSize()) > remote.len)
372 {
373 throw Exception::invalidArgument("Local region group is too large for the remote region.");
374 }
375
376 remoteRmaIov.addr = remote.addr + remoteOffset;
377 remoteRmaIov.len = localGroupSpan.byteSize();
378
379 auto const isLastWrite = (i == nbWrites - 1);
380 auto const immDataForWrite = isLastWrite ? immData : std::nullopt;
381
382 writeImpl(token,
383 localGroupSpan.asIovec(),
384 localGroupSpan.size(),
385 const_cast<void**>(localGroupSpan.desc()),
386 &remoteRmaIov,
387 destAddr,
388 immDataForWrite);
389
390 remoteOffset += localGroupSpan.byteSize();
391 }
392
393 return nbWrites;
394 }
