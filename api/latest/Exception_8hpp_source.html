145 template<typename F, typename... T>

147 {
148 int result = fun(std::forward<T>(args)...);
149 if (result < 0)
150 {
151 auto str = ::fi_strerror(result);
152 throw FabricException::make(result, "{}: {}, code {}", msg, str, result);
153 }
154

156 }
