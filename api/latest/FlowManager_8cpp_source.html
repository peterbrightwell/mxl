41 {

43 if (::mkdtemp(pathBuffer.data()) == nullptr)

54 bool publishFlowDirectory(std::filesystem::path const& source, std::filesystem::path const& dest)
55 {
56 permissions(source,
57 std::filesystem::perms::group_read | std::filesystem::perms::group_exec | std::filesystem::perms::others_read |
58 std::filesystem::perms::others_exec,
59 std::filesystem::perm_options::add);
60
61#if defined __linux__
62 if (::renameat2(AT_FDCWD, source.c_str(), AT_FDCWD, dest.c_str(), RENAME_NOREPLACE) < 0)
63#elif defined __APPLE__
64 if (::renamex_np(source.c_str(), dest.c_str(), RENAME_EXCL) < 0)
65#endif
66 {
67 auto const error = errno;
68 switch (error)
69 {
70 case EEXIST: return false;
71 case ENOTEMPTY: return false;
72 default:
73 throw std::system_error{error, std::system_category(), "Failed to publish flow directory by renaming it to its public name."};
74 }
75 }
76
77 return true;
78 }

165 std::pair<bool, std::unique_ptr<DiscreteFlowData>> FlowManager::createOrOpenDiscreteFlow(uuids::uuid const& flowId, std::string const& flowDef,
166 mxlDataFormat flowFormat, std::size_t grainCount, mxlRational const& grainRate, std::size_t grainPayloadSize, std::size_t grainNumOfSlices,
167 std::array<uint32_t, MXL_MAX_PLANES_PER_GRAIN> grainSliceLengths, std::uint32_t maxSyncBatchSizeHintOpt,
168 std::uint32_t maxCommitBatchSizeHintOpt)
169 {
170 auto const uuidString = uuids::to_string(flowId);
171 MXL_DEBUG("Create discrete flow. id: {}, grainCount: {}, grain payload size: {}", uuidString, grainCount, grainPayloadSize);
172
173 flowFormat = sanitizeFlowFormat(flowFormat);
174 if (!mxlIsDiscreteDataFormat(flowFormat))
175 {
176 throw std::runtime_error{"Attempt to create discrete flow with unsupported or non matching format."};
177 }
178
179 auto const tempDirectory = createTemporaryFlowDirectory(_mxlDomain);
180 auto _ = defer(
181 [&]() noexcept
182 {
183 std::error_code ec;
184 std::filesystem::remove_all(tempDirectory, ec);
185 if (ec)
186 {
187 MXL_WARN("Failed to remove temporary flow directory: {}", ec.message());
188 }
189 });
190
191 // Write the json file to disk.
192 writeFlowDescriptor(tempDirectory, flowDef);
193
194 // Create the dummy file.
195 auto readAccessFile = makeFlowAccessFilePath(tempDirectory);
196 if (auto out = std::ofstream{readAccessFile, std::ios::out | std::ios::trunc}; !out)
197 {
198 throw std::filesystem::filesystem_error{
199 "Failed to create flow access file.", readAccessFile, std::make_error_code(std::errc::file_exists)};
200 }
201
202 auto const flowDataPath = makeFlowDataFilePath(tempDirectory);
203 auto flowData = std::make_unique<DiscreteFlowData>(flowDataPath.string().c_str(), AccessMode::CREATE_READ_WRITE, LockMode::Shared);
204
205 auto& info = *flowData->flowInfo();
206 info.version = FLOW_DATA_VERSION;
207 info.size = sizeof info;
208 info.config.common = initCommonFlowConfigInfo(flowId, flowFormat, grainRate, maxSyncBatchSizeHintOpt, maxCommitBatchSizeHintOpt);
209 info.config.discrete = {};
210 info.config.discrete.grainCount = grainCount;
211 std::copy(grainSliceLengths.begin(), grainSliceLengths.end(), info.config.discrete.sliceSizes);
212
213 info.runtime = initFlowRuntimeInfo();
214
215 auto& state = *flowData->flowState();
216 state = initFlowState(flowDataPath);
217
218 auto const grainDir = makeGrainDirectoryName(tempDirectory);
219 if (!create_directory(grainDir))
220 {
221 throw std::filesystem::filesystem_error{"Could not create grain directory.", grainDir, std::make_error_code(std::errc::io_error)};
222 }
223
224 for (auto i = std::size_t{0}; i < grainCount; ++i)
225 {
226 auto const grainPath = makeGrainDataFilePath(grainDir, i);
227 MXL_TRACE("Creating grain: {}", grainPath.string());
228
229 // \todo Handle payload stored device memory
230 auto const grain = flowData->emplaceGrain(grainPath.string().c_str(), grainPayloadSize);
231 auto& gInfo = grain->header.info;
232 gInfo.grainSize = grainPayloadSize;
233 gInfo.totalSlices = grainNumOfSlices;
234 gInfo.validSlices = 0;
235 gInfo.version = GRAIN_HEADER_VERSION;
236 gInfo.size = sizeof gInfo;
237 }
238
239 auto const finalDir = makeFlowDirectoryName(_mxlDomain, uuidString);
240 if (publishFlowDirectory(tempDirectory, finalDir))
241 {
242 return {true, std::move(flowData)};
243 }
244 else
245 {
246 auto existingFlowData = dynamic_pointer_cast<DiscreteFlowData>(openFlow(flowId, AccessMode::READ_WRITE));
247 if (!existingFlowData)
248 {
249 throw std::runtime_error("Could not open existing flow because it is of a different format");
250 }
251
252 return {false, std::move(existingFlowData)};
253 }
254 }

321 std::unique_ptr<FlowData> FlowManager::openFlow(uuids::uuid const& in_flowId, AccessMode in_mode)
322 {
323 if (in_mode == AccessMode::CREATE_READ_WRITE)
324 {
325 throw std::invalid_argument{"Attempt to open flow with invalid access mode."};
326 }
327
328 auto uuid = uuids::to_string(in_flowId);
329 auto const base = makeFlowDirectoryName(_mxlDomain, uuid);
330
331 // Verify that the flow file exists.
332 if (auto const flowFile = makeFlowDataFilePath(base); exists(flowFile))
333 {
334 auto flowSegment = SharedMemoryInstance<Flow>{flowFile.string().c_str(), in_mode, 0U, LockMode::Shared};
335 if (flowSegment.get()->info.version != FLOW_DATA_VERSION)
336 {
337 throw std::invalid_argument{
338 fmt::format("Unsupported flow data version: {}, supported is: {}", flowSegment.get()->info.version, FLOW_DATA_VERSION)};
339 }
340
341 if (auto const flowFormat = flowSegment.get()->info.config.common.format; mxlIsDiscreteDataFormat(flowFormat))
342 {
343 return openDiscreteFlow(base, std::move(flowSegment));
344 }
345 else if (mxlIsContinuousDataFormat(flowFormat))
346 {
347 return openContinuousFlow(base, std::move(flowSegment));
348 }
349 else
350 {
351 // This should never happen for a valid flow.
352 throw std::runtime_error{"Attempt to open flow with unsupported data format."};
353 }
354 }
355 else
356 {
357 throw std::filesystem::filesystem_error{"Flow file not found.", flowFile, std::make_error_code(std::errc::no_such_file_or_directory)};
358 }
359 }

361 std::unique_ptr<DiscreteFlowData> FlowManager::openDiscreteFlow(std::filesystem::path const& flowDir,
362 SharedMemoryInstance<Flow>&& sharedFlowInstance)
363 {
364 auto flowData = std::make_unique<DiscreteFlowData>(std::move(sharedFlowInstance));
365
366 auto const grainCount = flowData->flowInfo()->config.discrete.grainCount;
367 if (grainCount > 0U)
368 {
369 auto const grainDir = makeGrainDirectoryName(flowDir);
370 if (exists(grainDir) && is_directory(grainDir))
371 {
372 // Open each grain with per-item error handling
373 for (auto i = 0U; i < grainCount; ++i)
374 {
375 auto const grainPath = makeGrainDataFilePath(grainDir, i).string();
376 MXL_TRACE("Opening grain: {}", grainPath);
377
378 flowData->emplaceGrain(grainPath.c_str(), /*payloadSize=*/0U);
379 }
380 }
381 else
382 {
383 throw std::filesystem::filesystem_error{
384 "Grain directory not found.", grainDir, std::make_error_code(std::errc::no_such_file_or_directory)};
385 }
386 }
387
388 return flowData;
389 }

445 {
446 auto base = std::filesystem::path{_mxlDomain};
447
448 auto flowIds = std::vector<uuids::uuid>{};
449 if (exists(base) && is_directory(base))
450 {
451 for (auto const& entry : std::filesystem::directory_iterator{_mxlDomain})
452 {
453 if (is_directory(entry) && (entry.path().extension() == FLOW_DIRECTORY_NAME_SUFFIX))

455 // this looks like a uuid. try to parse it an confirm it is valid.
456 auto id = uuids::uuid::from_string(entry.path().stem().string());

462 }

464 else
465 {
466 throw std::filesystem::filesystem_error{"Base directory not found.", base, std::make_error_code(std::errc::no_such_file_or_directory)};

472 std::string FlowManager::getFlowDef(uuids::uuid const& flowId) const
473 {
474 auto const uuid = uuids::to_string(flowId);
475 auto const flowPath = makeFlowDirectoryName(_mxlDomain, uuid);
476 auto const flowJsonFile = makeFlowDescriptorFilePath(flowPath);
477 if (auto in = std::ifstream{flowJsonFile, std::ios::in}; in)
478 {
479 auto const result = std::string{std::istreambuf_iterator<char>{in}, std::istreambuf_iterator<char>{}};
480 if (!in)
481 {
482 throw std::runtime_error{"Error while reading the flow definition."};
483 }
484 return result;
485 }
486 // Here is a race condition, but plain C++ API does not provide a way to check whether it was not possible to open a file because it does
487 // not exist, or whether the access rights are wrong.
488 if (!exists(flowJsonFile))
489 {
490 throw std::filesystem::filesystem_error{"Failed to open flow resource definition - file not found.",
491 flowJsonFile,
492 std::make_error_code(std::errc::no_such_file_or_directory)};
493 }
494 throw std::runtime_error{"Failed to open flow resource definition."};
495 }
