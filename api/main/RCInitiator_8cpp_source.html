46 bool RCInitiatorEndpoint::hasPendingWork() const noexcept
47 {
48 return std::visit(
49 overloaded{
50 [](std::monostate) { return false; }, // Something went wrong with this target, but there is probably no work to do.
51 [](Idle const&) { return true; }, // An idle target means there is no work to do right now.
52 [](Connecting const&) { return true; }, // In the connecting state, the target is waiting for a connected event.
53 [&](Connected const&) { return _proto->hasPendingWork(); }, // While connected, a target has pending work when
54 // there are transfers that have not yet completed.
55 [](Flushing const&) { return true; }, // In the shutdown state, the target is waiting for a FI_SHUTDOWN event.
56 [](Done const&) { return false; }, // In the done state, there is no pending work.
57 },
58 _state);
59 }

61 void RCInitiatorEndpoint::shutdown()
62 {
63 _state = std::visit(
64 overloaded{
65 [](Idle) -> State
66 {
67 MXL_INFO("Shutdown requested while waiting to activate, aborting.");
68 return Done{};
69 },
70 [](Connecting) -> State
71 {
72 MXL_INFO("Shutdown requested while trying to connect, aborting.");
73 return Done{};
74 },
75 [&](Connected state) -> State
76 {
77 MXL_INFO("Shutting down");
78 auto pending = _proto->reset();
79 state.ep.shutdown();
80
81 return Flushing{.ep = std::move(state.ep), .pending = pending};
82 },
83 [](Flushing state) -> State
84 {
85 MXL_WARN("Another shutdown was requested while trying to flush the completion queue, ignoring.");
86 return state;
87 },
88 [](Done state) -> State { return state; },
89 },
90 std::move(_state));
91 }

102 void RCInitiatorEndpoint::activate(std::shared_ptr<CompletionQueue> const& cq, std::shared_ptr<EventQueue> const& eq)
103 {
104 _state = std::visit(
105 overloaded{
106 [&](Idle state) -> State
107 {
108 // If the target is in the idle state for more than 5 seconds, it will be restarted.
109 auto idleDuration = std::chrono::steady_clock::now() - state.idleSince;
110 if (idleDuration < std::chrono::seconds(5))
111 {
112 return state;
113 }
114
115 MXL_INFO("Endpoint has been idle for {}ms, activating",
116 std::chrono::duration_cast<std::chrono::milliseconds>(idleDuration).count());
117
118 // The endpoint in an idle target is always fresh and thus needs to be bound the the queues.
119 state.ep.bind(eq);
120 state.ep.bind(cq, FI_TRANSMIT);
121
122 // Transition into the connecting state
123 state.ep.connect(_info.fabricAddress);
124 return Connecting{.ep = std::move(state.ep)};
125 },
126 [](Connecting state) -> State { return state; },
127 [](Connected state) -> State { return state; },
128 [](Flushing state) -> State { return state; },
129 [](Done state) -> State { return state; },
130 },
131 std::move(_state));
132 }

134 void RCInitiatorEndpoint::consume(Event ev)
135 {
136 _state = std::visit(
137 overloaded{
138 [](Idle state) -> State { return state; }, // Nothing to do when idle.
139 [&](Connecting state) -> State
140 {
141 if (ev.isError())
142 {
143 MXL_ERROR("Failed to connect endpoint: {}", ev.error().toString());
144
145 // Go into the idle state with a new endpoint
146 return restart(state.ep);
147 }
148 else if (ev.isConnected())
149 {
150 MXL_INFO("Endpoint is now connected");
151 return Connected{.ep = std::move(state.ep)};
152 }
153 else if (ev.isShutdown())
154 {
155 MXL_WARN("Received a shutdown event while connecting, going idle");
156
157 // Go to idle state with a new endpoint
158 return restart(state.ep);
159 }
160
161 MXL_WARN("Received an unexpected event while establishing a connection");
162 return state;
163 },
164 [&](Connected state) -> State
165 {
166 if (ev.isError())
167 {
168 MXL_WARN("Received an error event in connected state, going idle. Error: {}", ev.error().toString());
169 return restart(state.ep);
170 }
171 else if (ev.isShutdown())
172 {
173 MXL_INFO("Remote endpoint has closed the connection");
174 return Flushing{.ep = std::move(state.ep), .pending = _proto->reset()};
175 }
176
177 return state;
178 },
179 [&](Flushing state) -> State
180 {
181 if (ev.isShutdown())
182 {
183 MXL_INFO("Received a Shutdown Event while flushing the completion queue");
184 }
185 else if (ev.isError())
186 {
187 MXL_ERROR("Received an error while shutting down: {}", ev.error().toString());
188 return Done{};
189 }
190 else
191 {
192 MXL_ERROR("Received an unexpected event while shutting down");
193 }
194 return state;
195 },
196 [](Done state) -> State { return state; },
197 },
198 std::move(_state));
199 }

230 void RCInitiatorEndpoint::handleCompletionData(Completion::Data completion)
231 {
232 _state = std::visit(
233 overloaded{[](Idle state) -> State
234 {
235 MXL_WARN("Received a completion event while idle, ignoring.");
236 return state;
237 },
238 [](Connecting state) -> State
239 {
240 MXL_WARN("Received a completion event while connecting, ignoring.");
241 return state;
242 },
243 [&](Connected state) -> State
244 {
245 _proto->processCompletion(completion);
246 return state;
247 },
248 [](Flushing state) -> State
249 {
250 state.pending--; // Decrease the counter of completions that must still be flushed.
251 return state;
252 },
253 [](Done state) -> State
254 {
255 MXL_DEBUG("Ignoring completion after shutdown.");
256 return state;
257 }},
258 std::move(_state));
259 }

271 std::unique_ptr<RCInitiator> RCInitiator::setup(mxlFabricsInitiatorConfig const& config)
272 {
273 auto provider = providerFromAPI(config.interface.provider);
274 if (!provider)
275 {
276 throw Exception::make(MXL_ERR_NO_FABRIC, "No provider available");
277 }
278
279 uint64_t caps = FI_RMA | FI_WRITE | FI_REMOTE_WRITE;
280 // To enable device memory support:
281 // caps |= FI_HMEM;
282 auto fabricInfoList = FabricInfoList::get(config.interface.address.node, config.interface.address.service, provider.value(), caps, FI_EP_MSG);
283
284 if (fabricInfoList.begin() == fabricInfoList.end())
285 {
286 throw Exception::make(MXL_ERR_NO_FABRIC, "No suitable fabric available");
287 }
288
289 auto info = *fabricInfoList.begin();
290 MXL_DEBUG("{}", fi_tostr(info.raw(), FI_TYPE_INFO));
291
292 auto fabric = Fabric::open(info);
293 auto domain = Domain::open(fabric);
294
295 auto eq = EventQueue::open(fabric);
296 auto cq = CompletionQueue::open(domain);
297
298 auto regions = MxlRegions::forReader(config.reader);
299 auto proto = selectEgressProtocol(regions.dataLayout(), regions.regions());
300 proto->registerMemory(domain);

408 void RCInitiator::blockOnCQ(std::chrono::steady_clock::duration timeout)
409 {
410 // A zero timeout would cause the queue to block indefinetly, which
411 // is not our documented behaviour.
412 if (timeout == std::chrono::milliseconds::zero())
413 {
414 // So just behave exactly like the non-blocking variant.
415 makeProgress();
416 return;
417 }
418
419 for (;;)
420 {
421 auto completion = _cq->readBlocking(timeout);
422 if (!completion)
423 {
424 // No completion available, if we were flushing any endpoint, transition their state to done.
425 for (auto& [_, target] : _targets)
426 {
427 target.terminate();
428 }
429 return;
430 }
431
432 // Find the endpoint that this completion was generated from
433 auto ep = _targets.find(Endpoint::idFromToken(completion->token()));
434 if (ep == _targets.end())
435 {
436 MXL_WARN("Received completion for an unknown endpoint");
437 }
438
439 return ep->second.consume(*completion);
440 }
441 }

443 void RCInitiator::pollCQ()
444 {
445 for (;;)
446 {
447 auto completion = _cq->read();
448 if (!completion)
449 {
450 // No completion available, if we were flushing any endpoint, transition their state to done.
451 for (auto& [_, target] : _targets)
452 {
453 target.terminate();
454 }
455 break;
456 }
457
458 // Find the endpoint this completion was generated from.
459 auto ep = _targets.find(Endpoint::idFromToken(completion->token()));
460 if (ep == _targets.end())
461 {
462 MXL_WARN("Received completion for an unknown endpoint");
463
464 continue;
465 }
466
467 ep->second.consume(*completion);
468 }
469 }

494 bool RCInitiator::makeProgress()
495 {
496 if (!hasTarget())
497 {
498 throw Exception::interrupted("No more targets available while calling makeProgress.");
499 }
500
501 // Activate any peers that might be idle and waiting for activation.
502 activateIdleEndpoints();
503
504 // Poll the completion and event queue once and process pending events.
505 pollCQ();
506 pollEQ();
507
508 // Evict any peers that are dead and no longer will make progress.
509 evictDeadEndpoints();
510
511 return hasPendingWork();
512 }
513
514 bool RCInitiator::makeProgressBlocking(std::chrono::steady_clock::duration timeout)
515 {
516 // If the timeout is less than our maintainance interval, just check all the queues once, execute all maintainance tasks once
517 // and block on the completion queue for the rest of the time.
518 if (timeout < EQPollInterval)
519 {
520 makeProgress();
521 blockOnCQ(timeout);
522 return hasPendingWork();
523 }
524
525 auto deadline = std::chrono::steady_clock::now() + timeout;
526
527 for (;;)
528 {
529 // Poll all queues, execute all maintainance actions
530 if (!makeProgress())
531 {
532 return false;
533 }
534
535 // Calculate the remaining time until the user wants the blocking function to return. If there is no time left
536 // (millisecond precision) return right away.
537 auto timeUntilDeadline = std::chrono::duration_cast<std::chrono::milliseconds>(deadline - std::chrono::steady_clock::now());
538 if (timeUntilDeadline <= decltype(timeUntilDeadline){0})
539 {
540 return hasPendingWork();
541 }
542
543 // Block on the completion queue until a completion arrives, or the interval timeout occurs.
544 blockOnCQ(std::min(EQPollInterval, timeUntilDeadline));
545 }
546
547 return hasPendingWork();
548 }
