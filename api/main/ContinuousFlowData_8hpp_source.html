72 inline void ContinuousFlowData::openChannelBuffers(char const* grainFilePath, std::size_t sampleWordSize)
73 {
74 if ((sampleWordSize != 0U) || !created())
75 {
76 auto const info = flowInfo();
77 auto const channelCount = info->config.continuous.channelCount;
78 auto const bufferLength = info->config.continuous.bufferLength;
79 if (auto const buffersLength = channelCount * bufferLength; buffersLength > 0U)
80 {
81 auto const mode = this->created() ? AccessMode::CREATE_READ_WRITE : this->accessMode();
82 _channelBuffers = SharedMemorySegment{grainFilePath, mode, buffersLength * sampleWordSize, LockMode::Shared};
83
84 auto const mappedSize = _channelBuffers.mappedSize();
85 _sampleWordSize = (sampleWordSize != 0U) ? sampleWordSize : ((mappedSize >= buffersLength) ? (mappedSize / buffersLength) : 1U);
86 }
87 else
88 {
89 throw std::runtime_error{"Attempt to open channel buffer with invalid geometry."};
90 }
91 }
92 else
93 {
94 throw std::runtime_error{"Attempt to create channel buffer with invalid sample word size."};
95 }
96 }
