24 void InitiatorWrapper::setup(mxlFabricsInitiatorConfig const& config)
25 {
26 if (_inner)
27 {
28 _inner.reset();
29 }
30
31 switch (config.interface.provider)
32 {
33 case MXL_FABRICS_PROVIDER_ANY:
34 case MXL_FABRICS_PROVIDER_TCP:
35 case MXL_FABRICS_PROVIDER_VERBS:
36 {
37 _inner = RCInitiator::setup(config);
38 return;
39 }
40
41 case MXL_FABRICS_PROVIDER_SHM:
42 case MXL_FABRICS_PROVIDER_EFA:
43 {
44 _inner = RDMInitiator::setup(config);
45 return;
46 }
47 }
48
49 throw Exception::invalidArgument("Invalid provider value");
50 }
