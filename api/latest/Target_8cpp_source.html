78 std::unique_ptr<TargetInfo> TargetWrapper::setup(mxlFabricsTargetConfig const& config, TargetSetupOptions const& options)
79 {
80 if (_inner)
81 {
82 _inner.reset();
83 }
84
85 switch (config.interface.provider)
86 {
87 case MXL_FABRICS_PROVIDER_ANY: [[fallthrough]];
88 case MXL_FABRICS_PROVIDER_TCP: [[fallthrough]];
89 case MXL_FABRICS_PROVIDER_VERBS:
90 {
91 auto [target, info] = RCTarget::setup(config, options);
92 _inner = std::move(target);
93 return std::move(info);
94 }
95
96 case MXL_FABRICS_PROVIDER_SHM: [[fallthrough]];
97 case MXL_FABRICS_PROVIDER_EFA:
98 {
99 auto [target, info] = RDMTarget::setup(config, options);
100 _inner = std::move(target);
101 return std::move(info);
102 }
103 }
104
105 throw Exception::invalidArgument("Invalid provider value");
106 }
