27 std::string TargetInfo::toJSON() const
28 {
29 auto root = picojson::object{};
30
31 root["fabricAddress"] = picojson::value(fabricAddress.toBase64());
32
33 auto regions = picojson::array{};
34 for (auto const& remoteRegion : remoteRegions)
35 {
36 auto region = picojson::object{};
37 region["addr"] = picojson::value(std::to_string(remoteRegion.addr));
38 region["len"] = picojson::value(std::to_string(remoteRegion.len));
39 region["rkey"] = picojson::value(std::to_string(remoteRegion.rkey));
40 regions.emplace_back(region);
41 }
42 root["regions"] = picojson::value(regions);
43
44 if (bounceBufferInfo)
45 {
46 auto bounceBufferInfoObj = picojson::object{};
47 bounceBufferInfoObj["entryCount"] = picojson::value(std::to_string(bounceBufferInfo->entryCount));
48 bounceBufferInfoObj["entrySize"] = picojson::value(std::to_string(bounceBufferInfo->entrySize));
49 root["bounceBufferInfo"] = picojson::value(bounceBufferInfoObj);
50 }
51
52 root["id"] = picojson::value(std::to_string(id));
53
54 return picojson::value(root).serialize(false);
55 }

57 TargetInfo TargetInfo::fromJSON(std::string const& s)
58 {
59 auto parsed = picojson::value{};
60 auto const err = picojson::parse(parsed, s);
61 if (!err.empty())
62 {
63 throw Exception::invalidArgument("Failure when parsing JSON: {} ", err);
64 }
65
66 if (!parsed.is<picojson::object>())
67 {
68 throw Exception::invalidArgument("Expected a JSON object");
69 }
70
71 auto root = parsed.get<picojson::object>();
72
73 auto const fabricAddress = FabricAddress::fromBase64(root.at("fabricAddress").get<std::string>());
74
75 auto regions = std::vector<RemoteRegion>{};
76 for (auto const& regionValue : root.at("regions").get<picojson::array>())
77 {
78 auto const regionObj = regionValue.get<picojson::object>();
79
80 // Conversions of types that represent memory addresses to strings is preferred over converting to float
81 // to make sure nobody can reduce precision when converting on the way.
82 auto const addr = std::stoull(regionObj.at("addr").get<std::string>());
83 auto const len = std::stoull(regionObj.at("len").get<std::string>());
84 auto const rkey = std::stoull(regionObj.at("rkey").get<std::string>());
85 regions.emplace_back(addr, len, rkey);
86 }
87
88 auto bounceBufferInfo = std::optional<TargetInfoBounceBufferInfo>{std::nullopt};
89 if (root.contains("bounceBufferInfo"))
90 {
91 auto const bounceBufferInfoObj = root.at("bounceBufferInfo").get<picojson::object>();
92 auto const entryCount = std::stoull(bounceBufferInfoObj.at("entryCount").get<std::string>());
93 auto const entrySize = std::stoull(bounceBufferInfoObj.at("entrySize").get<std::string>());
94 bounceBufferInfo = std::make_optional<TargetInfoBounceBufferInfo>(entryCount, entrySize);
95 }
96
97 auto const id = std::stoull(root.at("id").get<std::string>());
98
99 return {.id = id, .fabricAddress = fabricAddress, .remoteRegions = regions, .bounceBufferInfo = bounceBufferInfo};
100 }
