67 template<typename T>
68 bool waitUntilChanged(T const* in_addr, T in_expected, Timepoint in_deadline)
69 {
70 static_assert(valid_specialization_v<T>, "Only 32 bit types with natural alignment are supported.");
71
72#if _LIBCPP_VERSION
73 // libc++ limitation due to: https://github.com/llvm/llvm-project/issues/118378
74 auto syncObject = std::atomic_ref{*const_cast<T*>(in_addr)};
75#else
76 auto syncObject = std::atomic_ref{*in_addr};
77#endif
78 while (syncObject.load(std::memory_order_acquire) == in_expected)
79 {
80 auto const now = currentTime(Clock::Realtime);
81 if (now >= in_deadline)
82 {
83 MXL_DEBUG("Deadline already reached");
84 return false;
85 }
86
87 // ATTENTION: Do not check for != 0, as positive values are success indicators
88 // in the macOS implementation of this function.
89 // NOTE: Unfortunately syncObject.address() is only available from C++26 onwards
90 if (auto const ret = do_wait(in_addr, in_expected, in_deadline - now); ret == -1)
91 {
92 switch (errno)
93 {
94 case EAGAIN:
95 // On linux EAGAIN indicates that the kernel detected *in_addr to
96 // not be equal to in_expected.
97 continue;
98
99 case EINTR:
100 // Interrupted. try again.
101 continue;
102
103 case ETIMEDOUT: MXL_TRACE("ETIMEDOUT. returning false"); break;
104
105 default: break;
106 }
107 return false;
108 }
109 }
110 return true;
111 }
