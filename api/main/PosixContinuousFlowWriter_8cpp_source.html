74 mxlStatus PosixContinuousFlowWriter::openSamples(std::uint64_t index, std::size_t count, mxlMutableWrappedMultiBufferSlice& payloadBufferSlices)
75 {
76 if (_flowData)
77 {
78 if (count <= (_bufferLength / 2))
79 {
80 auto const startOffset = (index + _bufferLength - count) % _bufferLength;
81 auto const endOffset = (index % _bufferLength);
82
83 auto const firstLength = (startOffset < endOffset) ? count : _bufferLength - startOffset;
84 auto const secondLength = count - firstLength;
85
86 auto const baseBufferPtr = static_cast<std::uint8_t*>(_flowData->channelData());
87 auto const sampleWordSize = _flowData->sampleWordSize();
88
89 payloadBufferSlices.base.fragments[0].pointer = baseBufferPtr + sampleWordSize * startOffset;
90 payloadBufferSlices.base.fragments[0].size = sampleWordSize * firstLength;
91
92 payloadBufferSlices.base.fragments[1].pointer = baseBufferPtr;
93 payloadBufferSlices.base.fragments[1].size = sampleWordSize * secondLength;
94
95 payloadBufferSlices.stride = sampleWordSize * _bufferLength;
96 payloadBufferSlices.count = _channelCount;
97
98 _currentIndex = index;
99
100 return MXL_STATUS_OK;
101 }
102
103 return MXL_ERR_INVALID_ARG;
104 }
105 return MXL_ERR_UNKNOWN;
106 }

108 mxlStatus PosixContinuousFlowWriter::commit()
109 {
110 if (_flowData)
111 {
112 auto const flow = _flowData->flow();
113 flow->info.runtime.headIndex = _currentIndex;
114 _currentIndex = MXL_UNDEFINED_INDEX;
115
116 if (signalCompletedBatch())
117 {
118 // Let readers know that the head has moved
119 flow->state.syncCounter++;
120 wakeAll(&flow->state.syncCounter);
121 }
122
123 return MXL_STATUS_OK;
124 }
125 else
126 {
127 _currentIndex = MXL_UNDEFINED_INDEX;
128 return MXL_ERR_UNKNOWN;
129 }

138 bool PosixContinuousFlowWriter::signalCompletedBatch() noexcept
139 {
140 auto const currentSyncSampleBatch = _currentIndex / _syncBatchSize;
141 if (currentSyncSampleBatch < _lastSyncSampleBatch)
142 {
143 return false;
144 }
145 if (currentSyncSampleBatch == _lastSyncSampleBatch)
146 {
147 // Signal now before overshooting the maximum the next time around
148 if ((_currentIndex % _syncBatchSize) > _earlySyncThreshold)
149 {
150 _lastSyncSampleBatch = currentSyncSampleBatch + 1U;
151 }
152 else
153 {
154 return false;
155 }
156 }
157 else
158 {
159 _lastSyncSampleBatch = currentSyncSampleBatch;
160 }
161
162 return true;
163 }
