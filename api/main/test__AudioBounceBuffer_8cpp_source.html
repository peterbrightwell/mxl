18 // 2 audio channels, each target channel has a ring buffer of 9 samples and we are unpacking 5 samples landing at head index 11.
19 // The start offset is (11 + 9 - 5) % 9 = 6 so each channel writes to target indices:
20 // 6, 7, 8, 0, 1
21 // which creates a split of 3 samples before wrap + 2 samples after wrap
22 constexpr auto ChannelCount = std::size_t{2};
23 constexpr auto TargetBufferLength = std::size_t{9};
24 constexpr auto HeadIndex = std::uint64_t{11};
25 constexpr auto SampleCount = std::size_t{5};
26 constexpr auto UntouchedSample = Sample{0xFFFFFFFFU};

44TEST_CASE("ofi: audio bounce buffer unpacks channel-major payload across ring wrap", "[ofi][AudioBounceBuffer]")
45{
46 auto target = std::vector<Sample>(ChannelCount * TargetBufferLength, UntouchedSample);
47 auto targetSlice = mxlMutableWrappedMultiBufferSlice{};
48 AudioBounceBuffer::getMutableMultiBufferSlices(
49 HeadIndex, SampleCount, TargetBufferLength, sizeof(Sample), ChannelCount, reinterpret_cast<std::uint8_t*>(target.data()), targetSlice);
50
51 REQUIRE((targetSlice.base.fragments[0].size / sizeof(Sample)) == 3U);
52 REQUIRE((targetSlice.base.fragments[1].size / sizeof(Sample)) == 2U);
53
54 // channel 0 payload: 103, 104, 105, 106, 107
55 // channel 1 payload: 203, 204, 205, 206, 207
56 auto const payloadSamples = std::vector<Sample>{103U, 104U, 105U, 106U, 107U, 203U, 204U, 205U, 206U, 207U};
57 auto const payloadSize = payloadSamples.size() * sizeof(Sample);
58
59 auto const targetLayout = DataLayout::Continuous{.sampleSize = sizeof(Sample), .channelCount = ChannelCount, .bufferLength = TargetBufferLength};
60 auto bounceBuffer = AudioBounceBuffer{1, payloadSize, targetLayout};
61 auto bounceRegions = bounceBuffer.getRegions();
62 REQUIRE(bounceRegions.size() == 1U);
63 REQUIRE(bounceRegions.front().size == (sizeof(AudioEntryHeader) + payloadSize));
64
65 // Write directly into the bounce-buffer entry:
66 auto* entryData = reinterpret_cast<std::uint8_t*>(bounceRegions.front().base); // NOLINT
67 auto const header = AudioEntryHeader{.headIndex = HeadIndex, .count = SampleCount};
68 std::memcpy(entryData, &header, sizeof(header));
69 std::memcpy(entryData + sizeof(AudioEntryHeader), payloadSamples.data(), payloadSize);
70
71 // Let entry 0 of the bounce buffer copy its payload into the target ring buffer.
72 auto const outRegion = Region{regionBaseOf(target.data()), target.size() * sizeof(Sample), nullptr, nullptr, Region::Location::host()};
73 auto const unpackedHeader = bounceBuffer.unpack(0, outRegion);
74 REQUIRE(unpackedHeader.headIndex == HeadIndex);
75 REQUIRE(unpackedHeader.count == SampleCount);
76
77 // Verify that unpack did correctly consume channel-major payload
78 auto expectedTarget = std::vector<Sample>(ChannelCount * TargetBufferLength, UntouchedSample);
79 auto const targetStartOffset = ringStartOffset(HeadIndex, SampleCount, TargetBufferLength);
80 for (auto channel = std::size_t{0}; channel < ChannelCount; channel++)
81 {
82 for (auto offset = std::size_t{0}; offset < SampleCount; offset++)
83 {
84 auto const targetIndex = (targetStartOffset + offset) % TargetBufferLength;
85 sampleAt(expectedTarget, channel, targetIndex, TargetBufferLength) = payloadSamples[(channel * SampleCount) + offset];
86 }
87 }
88
89 REQUIRE(target == expectedTarget);
90}
