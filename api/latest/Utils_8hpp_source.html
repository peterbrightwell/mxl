78 inline std::string readFile(std::filesystem::path const& filepath)
79 {
80 if (auto file = std::ifstream{filepath, std::ios::in | std::ios::binary}; file)
81 {
82 auto reader = std::stringstream{};
83 reader << file.rdbuf();
84 return reader.str();
85 }
86
87 throw std::runtime_error("Failed to open file: " + filepath.string());
88 }

90 inline std::filesystem::path getDomainPath()
91 {
92#ifdef __linux__
93 return std::filesystem::path{"/dev/shm/mxl_domain"};
94#elif __APPLE__
95 if (auto const home = std::getenv("HOME"); home != nullptr)
96 {
97 return std::filesystem::path{home} / "mxl_domain";
98 }
99 else
100 {
101 throw std::runtime_error{"Environment variable HOME is not set."};
102 }
103#else
104# error "Unsupported platform. This is only implemented for Linux and macOS."
105#endif
106 }

108 inline std::filesystem::path makeTempDomain()
109 {
110#ifdef __linux__
111 char tmpl[] = "/dev/shm/mxl_test_domainXXXXXX";
112#elif __APPLE__
113 char tmpl[] = "/tmp/mxl_test_domainXXXXXX";
114#else
115# error "Unsupported platform. This is only implemented for Linux and macOS."
116#endif
117 if (::mkdtemp(tmpl) == nullptr)
118 {
119 throw std::runtime_error("Failed to create temporary directory");
120 }
121 return std::filesystem::path{tmpl};
122 }
