111 std::optional<Event> EventQueue::handleReadResult(ssize_t ret, std::uint32_t eventType, ::fi_eq_entry const& entry)
112 {
113 if (ret == -FI_EAGAIN)
114 {
115 // No event available
116 return std::nullopt;
117 }
118
119 if (ret == -FI_EAVAIL)
120 {
121 ::fi_eq_err_entry eq_err{};
122 fi_eq_readerr(_raw, &eq_err, 0);
123
124 return Event::fromError(this->shared_from_this(), &eq_err);
125 }
126
127 if (ret < 0)
128 {
129 throw FabricException::make(ret, "Failed to read from event queue: {}", ::fi_strerror(ret));
130 }
131
132 switch (eventType)
133 {
134 case FI_CONNECTED:
135 case FI_CONNREQ:
136 case FI_SHUTDOWN: return Event::fromRawCMEntry(*reinterpret_cast<fi_eq_cm_entry const*>(&entry), eventType); break;
137 case FI_NOTIFY:
138 case FI_MR_COMPLETE:
139 case FI_AV_COMPLETE:
140 case FI_JOIN_COMPLETE: return Event::fromRawEntry(entry, eventType); break;
141 default: MXL_WARN("Unhandled event on event queue: code: {}", eventType); return std::nullopt;
142 }
143 }
