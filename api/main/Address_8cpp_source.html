60 FabricAddress FabricAddress::retrieveFabricAddress(::fid_t fid)
61 {
62 // First obtain the address length
63 std::size_t addrlen = 0;
64 auto ret = fi_getname(fid, nullptr, &addrlen);
65 if (ret != -FI_ETOOSMALL)
66 {
67 throw FabricException("Failed to get address length from endpoint.", MXL_ERR_UNKNOWN, ret);
68 }
69
70 // Now that we have the address length, allocate a receiving buffer and call fi_getname again to retrieve the actual address
71 std::vector<std::uint8_t> addr(addrlen);
72 fiCall(fi_getname, "Failed to retrieve endpoint's local address.", fid, addr.data(), &addrlen);
73
74 return FabricAddress{addr};
75 }
