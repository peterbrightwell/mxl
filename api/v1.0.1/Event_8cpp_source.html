84 Event Event::fromRawCMEntry(::fi_eq_cm_entry const& entry, std::uint32_t eventType)
85 {
86 // clang-format off
87 switch (eventType)
88 {
89 case FI_CONNREQ: return {ConnectionRequested{entry.fid, FabricInfo::own(entry.info)}};
90 case FI_CONNECTED: return {Connected{entry.fid}};
91 case FI_SHUTDOWN: return {Shutdown{entry.fid}};
92 default: throw Exception::internal("Unsupported event type returned from queue");
93 }
94 // clang-format on
95 }
