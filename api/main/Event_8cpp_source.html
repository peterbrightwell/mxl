71 std::string Event::Error::toString() const
72 {
73 // _err is the libfabric-level error (FI_E*); _providerErr is set by the
74 // underlying provider (e.g. the tcp provider's errno). Surface both so
75 // callers see the real reason even when the provider has nothing extra
76 // to say (in which case fi_eq_strerror would otherwise return "Success").
77 auto const* fiStr = ::fi_strerror(_err);
78 auto const* provStr = ::fi_eq_strerror(_eq->raw(), _providerErr, _errData.data(), nullptr, 0);
79 return fmt::format("fi_err={} ({}) prov_err={} ({})", _err, fiStr ? fiStr : "", _providerErr, provStr ? provStr : "");
80 }

91 Event Event::fromRawCMEntry(::fi_eq_cm_entry const& entry, std::uint32_t eventType)
92 {
93 // clang-format off
94 switch (eventType)
95 {
96 case FI_CONNREQ: return {ConnectionRequested{entry.fid, FabricInfo::own(entry.info)}};
97 case FI_CONNECTED: return {Connected{entry.fid}};
98 case FI_SHUTDOWN: return {Shutdown{entry.fid}};
99 default: throw Exception::internal("Unsupported event type returned from queue");
100 }
101 // clang-format on
102 }

105 {
