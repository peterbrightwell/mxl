82 mxlStatus PosixDiscreteFlowWriter::openGrain(std::uint64_t in_index, mxlGrainInfo* out_grainInfo, std::uint8_t** out_payload)
83 {
84 if (_flowData)
85 {
86 auto offset = in_index % _flowData->flowInfo()->config.discrete.grainCount;
87 auto const grain = _flowData->grainAt(offset);
88 grain->header.info.index = in_index; // Set the absolute grain index associated to that ring buffer entry
89 *out_grainInfo = grain->header.info;
90 *out_payload = reinterpret_cast<std::uint8_t*>(&grain->header + 1);
91 _currentIndex = in_index;
92 return MXL_STATUS_OK;
93 }
94 return MXL_ERR_UNKNOWN;
95 }

123 mxlStatus PosixDiscreteFlowWriter::commit(mxlGrainInfo const& mxlGrainInfo)
124 {
125 if (_flowData)
126 {
127 if (mxlGrainInfo.index != _currentIndex)
128 {
129 return MXL_ERR_INVALID_ARG;
130 }
131
132 auto const flow = _flowData->flow();
133 flow->info.runtime.headIndex = _currentIndex;
134
135 auto const offset = _currentIndex % flow->info.config.discrete.grainCount;
136 *_flowData->grainInfoAt(offset) = mxlGrainInfo;
137 flow->info.runtime.lastWriteTime = currentTime(mxl::lib::Clock::TAI).value;
138
139 // If the grain is complete, reset the current index of the flow writer.
140 if (mxlGrainInfo.validSlices == mxlGrainInfo.totalSlices)
141 {
142 _currentIndex = MXL_UNDEFINED_INDEX;
143 }
144
145 // Let readers know that the head has moved or that new data is available in a partial grain
146 flow->state.syncCounter++;
147 wakeAll(&flow->state.syncCounter);
148
149 return MXL_STATUS_OK;
150 }
151 return MXL_ERR_UNKNOWN;
152 }
