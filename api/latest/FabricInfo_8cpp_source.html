161 FabricInfoList FabricInfoList::get(char const* node, char const* service, Provider provider, std::uint64_t caps, ::fi_ep_type epType)
162 {
163 ::fi_info* info;
164 auto hints = FabricInfo::empty();
165
167 hints->domain_attr->mr_mode = FI_MR_LOCAL | FI_MR_VIRT_ADDR | FI_MR_ALLOCATED | FI_MR_PROV_KEY | FI_MR_HMEM;
168
169 hints->mode = 0;
170 hints->caps = caps;
171 hints->ep_attr->type = epType;
172 hints->fabric_attr->prov_name = strdup(fmt::to_string(provider).c_str());
173
174 // hints: add condition to append FI_HMEM capability if needed!
175
176 fiCall(::fi_getinfo, "Failed to get provider information", fiVersion(), node, service, FI_SOURCE, hints.raw(), &info);
177
178 return FabricInfoList{info};
179 }
