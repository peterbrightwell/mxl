47 std::optional<Target::ReadResult> RMAGrainIngressProtocol::read(Endpoint const& endpoint, Completion const& completion)
48 {
49 auto completionData = completion.tryData();
50 if (!completionData)
51 {
52 return {};
53 }
54
55 if (_immDataBuffer)
56 {
57 endpoint.recv(_immDataBuffer->toLocalRegion());
58 }
59
60 auto immData = completionData->data();
61 if (!immData)
62 {
63 throw Exception::invalidState("Received a completion without immediate data.");
64 }
65
66 auto [slot, slice] = ImmDataGrain{static_cast<std::uint32_t>(*immData)}.unpack();
67
68 // Set the number of valid slices in the grain header. This information is received through the immediate data and must be updated
69 // in the local shared memory in the case of partial writes.
70 setValidSlicesForGrain(_regions, slot, slice);
71
72 // Get the actual grain index from the grain header in share memory. This was written in the first RMA write.
73 auto grainIndex = getGrainIndexInRingSlot(_regions, slot);
74
75 return std::make_optional<Target::GrainReadResult>(grainIndex);
76 }

183 AudioBounceBuffer RMASampleIngressProtocol::makeAudioBounceBuffer(DataLayout::Continuous const& layout, std::uint32_t maxSyncBatchSize)
184 {
185 auto const oneSampleSize = layout.sampleSize * layout.channelCount;
186 auto const entrySize = oneSampleSize * maxSyncBatchSize;
187 auto const historySize = layout.bufferLength * oneSampleSize;
188
189 auto const entryCount = (historySize + entrySize - 1U) / entrySize; // ceil(historySize / entrySize)
190
191 MXL_INFO("Creating audio bounce buffer with entry size {} bytes and entry count {}, maxSyncBatchSize {} historySize {} bufferLength {}",
192 entrySize,
193 entryCount,
194 maxSyncBatchSize,
195 historySize,
196 layout.bufferLength);
197 return {entryCount, entrySize, layout};
198 }
